191     /**
192      * Creates a instance of a <code>Configuration</code> object, which is
193      * initialized with the settings from the system-wide configuration file.
194      * 
195      * @param configSchemaFileName
196      *            The file name of the schema to check the configuration file
197      *            against.s
198      * @param configFileName
199      *            The file name of the configuration file to use.
200      * @return A <code>Configuration</code> instance with all settings.
201      * @throws ConfigurationException
202      *             If this operation is supported but failed for some reason.

205     public static final Configuration create(final File configSchemaFileName, final File configFileName)
206         throws ConfigurationException {
207 
208         final Configuration config = new Configuration();
209 
210         final Document doc = config.parse(configSchemaFileName, configFileName);
211         config.parseSettings(doc.getDocumentElement());
212 
213         return config;
214     }

216     /**
217      * Returns the value of a single parameter, instead of all values.
218      * 
219      * @param targetName
220      *            Name of the iSCSI Target to connect.
221      * @param connectionID
222      *            The ID of the connection to retrieve.
223      * @param textKey
224      *            The name of the parameter.
225      * @return The value of the given parameter.
226      * @throws OperationalTextKeyException
227      *             If the given parameter cannot be found.
228      */
229     public final String getSetting(final String targetName, final int connectionID,
230         final OperationalTextKey textKey) throws OperationalTextKeyException {
231 
232         try {
233             final SessionConfiguration sc;
234             synchronized (sessionConfigs) {
235                 sc = sessionConfigs.get(targetName);
236 
237                 synchronized (sc) {
238                     if (sc != null) {
239                         String value = sc.getSetting(connectionID, textKey);
240                         if (value != null) {
241                             return value;
242                         }
243                     }
244                 }
245             }
246         } catch (OperationalTextKeyException e) {
247             // we had not find a session/connection entry, so we have to search
248             // in the
249             // global settings
250         }
251 
252         final SettingEntry se;
253         synchronized (globalConfig) {
254             se = globalConfig.get(textKey);
255 
256             synchronized (se) {
257                 if (se != null) {
258                     return se.getValue();
259                 }
260             }
261         }
262 
263         throw new OperationalTextKeyException("No OperationalTextKey entry found for key: " + textKey.value());
264     }

266     /**
267      * Unifies all parameters (in the right precedence) and returns one <code>SettingsMap</code>. Right order
268      * means: default, then the
269      * session-wide, and finally the connection-wide valid parameters.
270      * 
271      * @param targetName
272      *            Name of the iSCSI Target to connect.
273      * @param connectionID
274      *            The ID of the connection to retrieve.
275      * @return All unified parameters in one single <code>SettingsMap</code>.
276      */
277     public final SettingsMap getSettings(final String targetName, final int connectionID) {
278 
279         final SettingsMap sm = new SettingsMap();
280 
281         // set all default settings
282         synchronized (globalConfig) {
283             for (Map.Entry<OperationalTextKey, SettingEntry> e : globalConfig.entrySet()) {
284                 sm.add(e.getKey(), e.getValue().getValue());
285             }
286         }
287 
288         // set all further settings
289         final SessionConfiguration sc;
290         synchronized (sessionConfigs) {
291             sc = sessionConfigs.get(targetName);
292 
293             synchronized (sc) {
294                 if (sc != null) {
295                     final SettingsMap furtherSettings = sc.getSettings(connectionID);
296                     for (Map.Entry<OperationalTextKey, String> e : furtherSettings.entrySet()) {
297                         sm.add(e.getKey(), e.getValue());
298                     }
299                 }
300             }
301         }
302 
303         return sm;
304     }

347     /**
348      * Updates the stored settings of a connection with these values from the
349      * response of the iSCSI Target.
350      * 
351      * @param targetName
352      *            The name of the iSCSI Target.
353      * @param connectionID
354      *            The ID of the connection within this iSCSI Target.
355      * @param response
356      *            The response settings.
357      * @throws NoSuchSessionException
358      *             if a session with this target name is not open.
359      */
360     public final void update(final String targetName, final int connectionID, final SettingsMap response)
361         throws NoSuchSessionException {
362 
363         final SessionConfiguration sc;
364         synchronized (sessionConfigs) {
365             sc = sessionConfigs.get(targetName);
366 
367             synchronized (sc) {
368                 if (sc == null) {
369                     throw new NoSuchSessionException("A session with the ID '" + targetName
370                         + "' does not exist.");
371                 }
372 
373                 synchronized (response) {
374                     SettingEntry se;
375                     for (Map.Entry<OperationalTextKey, String> e : response.entrySet()) {
376                         synchronized (globalConfig) {
377                             se = globalConfig.get(e.getKey());
378 
379                             if (se == null) {
380                                 if (LOGGER.isWarnEnabled()) {
381                                     LOGGER.warn("This key " + e.getKey() + " is not in the globalConfig.");
382                                 }
383                                 continue;
384                             }
385 
386                             synchronized (se) {
387                                 if (se.getScope().compareTo(VALUE_SCOPE_SESSION) == 0) {
388                                     sc.updateSessionSetting(e.getKey(), e.getValue(), se.getResult());
389                                 } else if (se.getScope().compareTo(VALUE_SCOPE_CONNECTION) == 0) {
390                                     sc.updateConnectionSetting(connectionID, e.getKey(), e.getValue(), se
391                                         .getResult());
392                                 }
393                             }
394                         }
395                     }
396                 }
397             }
398         }
399     }

404     /**
405      * Reads the given configuration file in memory and creates a DOM
406      * representation.
407      * 
408      * @throws SAXException
409      *             If this operation is supported but failed for some reason.
410      * @throws ParserConfigurationException
411      *             If a <code>DocumentBuilder</code> cannot be created which
412      *             satisfies the configuration requested.
413      * @throws IOException
414      *             If any IO errors occur.
415      */
416     private final Document parse(final File schemaLocation, final File configFile)
417         throws ConfigurationException {
418         try {
419             final SchemaFactory schemaFactory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
420             final Schema schema = schemaFactory.newSchema(schemaLocation);
421 
422             // create a validator for the document
423             final Validator validator = schema.newValidator();
424 
425             final DocumentBuilderFactory domFactory = DocumentBuilderFactory.newInstance();
426             domFactory.setNamespaceAware(true); // never forget this
427             final DocumentBuilder builder = domFactory.newDocumentBuilder();
428             final Document doc = builder.parse(configFile);
429 
430             final DOMSource source = new DOMSource(doc);
431             final DOMResult result = new DOMResult();
432 
433             validator.validate(source, result);
434 
435             return (Document)result.getNode();
436         } catch (final SAXException exc) {
437             throw new ConfigurationException(exc);
438         } catch (final ParserConfigurationException exc) {
439             throw new ConfigurationException(exc);
440         } catch (final IOException exc) {
441             throw new ConfigurationException(exc);
442         }
443     }

445     /**
446      * Parses all settings form the main configuration file.
447      * 
448      * @param root
449      *            The root element of the configuration.
450      */
451     private final void parseSettings(final Element root) {
452 
453         if (root == null) {
454             throw new NullPointerException();
455         }
456 
457         clear();
458         parseGlobalSettings(root);
459         parseTargetSpecificSettings(root);
460     }

462     /**
463      * Parses all global settings form the main configuration file.
464      * 
465      * @param root
466      *            The root element of the configuration.
467      */
468     private final void parseGlobalSettings(final Element root) {
469 
470         final NodeList globalConfiguration = root.getElementsByTagName(ELEMENT_GLOBAL);
471 
472         final ResultFunctionFactory resultFunctionFactory = new ResultFunctionFactory();
473         Node parameter;
474         NodeList parameters;
475         NamedNodeMap attributes;
476         SettingEntry key;
477         for (int i = 0; i < globalConfiguration.getLength(); i++) {
478             parameters = globalConfiguration.item(i).getChildNodes();
479 
480             for (int j = 0; j < parameters.getLength(); j++) {
481                 parameter = parameters.item(j);
482 
483                 if (parameter.getNodeType() == Node.ELEMENT_NODE) {
484                     attributes = parameter.getAttributes();
485 
486                     key = new SettingEntry();
487                     key.setScope(attributes.getNamedItem(ATTRIBUTE_SCOPE).getNodeValue());
488                     key.setResult(resultFunctionFactory.create(attributes.getNamedItem(ATTRIBUTE_RESULT)
489                         .getNodeValue()));
490                     // key.setSender(attributes.getNamedItem(ATTRIBUTE_SENDER).getNodeValue
491                     // ());
492                     key.setValue(parameter.getTextContent());
493 
494                     synchronized (globalConfig) {
495                         globalConfig.put(OperationalTextKey.valueOfEx(parameter.getNodeName()), key);
496                     }
497                 }
498             }
499         }
500     }
501 
502     /**
503      * Parses all target-specific settings form the main configuration file.
504      * 
505      * @param root
506      *            The root element of the configuration.
507      */
508     private final void parseTargetSpecificSettings(final Element root) {
509 
510         final NodeList targets = root.getElementsByTagName(ELEMENT_TARGET);
511 
512         Node target;
513         Node parameter;
514         NodeList parameters;
515 
516         try {
517             for (int i = 0; i < targets.getLength(); i++) {
518                 target = targets.item(i);
519                 parameters = target.getChildNodes();
520 
521                 // extract target address and the port (if specified)
522                 SessionConfiguration sc = new SessionConfiguration();
523 
524                 sc.setAddress(target.getAttributes().getNamedItem(ATTRIBUTE_ADDRESS).getNodeValue(), Integer
525                     .parseInt(target.getAttributes().getNamedItem(ATTRIBUTE_PORT).getNodeValue()));
526 
527                 // extract the parameters for this target
528                 for (int j = 0; j < parameters.getLength(); j++) {
529                     parameter = parameters.item(j);
530 
531                     if (parameter.getNodeType() == Node.ELEMENT_NODE) {
532                         sc.addSessionSetting(OperationalTextKey.valueOfEx(parameter.getNodeName()), parameter
533                             .getTextContent());
534                     }
535 
536                 }
537 
538                 synchronized (sessionConfigs) {
539                     sessionConfigs.put(target.getAttributes().getNamedItem(ATTRIBUTE_ID).getNodeValue(), sc);
540                 }
541             }
542         } catch (UnknownHostException e) {
543             if (LOGGER.isErrorEnabled()) {
544                 LOGGER.error("The given host is not reachable: " + e.getLocalizedMessage());
545             }
546         }
547     }
