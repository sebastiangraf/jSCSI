42  /**
43   * <h1>GetConnectionsRequestState</h1>
44   * <p/>
45   * This state requests a list of all possible connections to a specific target. So it sends a TextRequest PDU
46   * with <code>SendTargets=</code> as the only <code>OperationalTextKey</code> as data segment.
47   * 
48   * @author Volker Wildi
49   */
50  public final class GetConnectionsRequestState extends AbstractState {
51  
52      // --------------------------------------------------------------------------
53      // --------------------------------------------------------------------------
54  
55      /**
56       * Constructor to create a <code>GetConnectionsRequestState</code> instance,
57       * which uses the given connection for transmission.
58       * 
59       * @param initConnection
60       *            The context connection, which is used for the network
61       *            transmission.
62       */
63      public GetConnectionsRequestState(final Connection initConnection) {
64  
65          super(initConnection);
66      }
67  
68      // --------------------------------------------------------------------------
69      // --------------------------------------------------------------------------
70  
71      /** {@inheritDoc} */
72      public final void execute() throws InternetSCSIException {
73  
74          final ProtocolDataUnit protocolDataUnit =
75              protocolDataUnitFactory.create(false, true, OperationCode.TEXT_REQUEST, connection
76                  .getSetting(OperationalTextKey.HEADER_DIGEST), connection
77                  .getSetting(OperationalTextKey.DATA_DIGEST));
78          final TextRequestParser parser =
79              (TextRequestParser)protocolDataUnit.getBasicHeaderSegment().getParser();
80  
81          final SettingsMap settings = new SettingsMap();
82          settings.add(OperationalTextKey.SEND_TARGETS, "");
83  
84          final IDataSegment dataSegment =
85              DataSegmentFactory.create(settings.asByteBuffer(), DataSegmentFormat.TEXT, connection
86                  .getSettingAsInt(OperationalTextKey.MAX_RECV_DATA_SEGMENT_LENGTH));
87  
88          int bytes2Process = dataSegment.getLength();
89          for (IDataSegmentIterator dataSegmentIterator = dataSegment.iterator(); dataSegmentIterator.hasNext();) {
90              IDataSegmentChunk dataSegmentChunk = dataSegmentIterator.next(bytes2Process);
91              protocolDataUnit.setDataSegment(dataSegmentChunk);
92              parser.setTargetTransferTag(0xFFFFFFFF);
93          }
94  
95          connection.send(protocolDataUnit);
96          connection.nextState(new GetConnectionsResponseState(connection));
97          super.stateFollowing = true;
98          // return true;
99      }

103     // --------------------------------------------------------------------------
104     // --------------------------------------------------------------------------

106 }
