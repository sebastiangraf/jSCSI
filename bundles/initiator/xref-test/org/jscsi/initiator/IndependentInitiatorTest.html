87      /**
88       * A helping class to start the target in a threadpool so the tests can run
89       * next to it.
90       * 
91       */
92      public static class CallableStart {
93          static ExecutorService threadPool;
94  
95          /**
96           * This method starts a targetserver in a threadpool.
97           * 
98           * @throws SAXException
99           * @throws ParserConfigurationException
100          * @throws IOException
101          * @throws ConfigurationException
102          */
103         public static void start() throws SAXException, ParserConfigurationException, IOException,
104             ConfigurationException {
105             if (isWindows()) {
106                 targetConfigurationFile = new File(CONFIG_DIR, "jscsi-target-windows.xml");
107             } else {
108                 targetConfigurationFile = new File(CONFIG_DIR, "jscsi-target-linux.xml");
109             }
110 
111             org.jscsi.target.Configuration targetConfiguration =
112                 org.jscsi.target.Configuration.create(new File(CONFIG_DIR, "jscsi-target.xsd"),
113                     targetConfigurationFile);
114 
115             target = new TargetServer(targetConfiguration);
116 
117             // Getting an Executor
118             threadPool = Executors.newSingleThreadExecutor();
119             // Starting the target
120             threadPool.submit(target);
121         }
122 
123         /**
124          * After the tests are finished shutdown the threadpool.
125          */
126         public static void stop() {
127             threadPool.shutdown();
128         }
129     }

162      * This test checks if a written bytebuffer still is the same when being
163      * read from the target.
164      * 
165      * @throws Exception
166      */
167     @Test
168     public final void testReadWriteEquality() throws Exception {
169         Future<Void> returnVal1 = session.write(writeBuffer, LOGICAL_BLOCK_ADDRESS, writeBuffer.remaining());
170         returnVal1.get();
171 
172         assertTrue(!writeBuffer.equals(readBuffer));
173 
174         Future<Void> returnVal2 = session.read(readBuffer, LOGICAL_BLOCK_ADDRESS, readBuffer.remaining());
175         returnVal2.get();
176 
177         assertTrue(writeBuffer.equals(readBuffer));
178 
179         writeBuffer.flip();
180         readBuffer.flip();
181 
182         assertTrue(writeBuffer.equals(readBuffer));
183 
184     }

186     /**
187      * This test writes multiple buffers into the target and reads them
188      * afterwards.
189      */
190     @Test
191     public final void testMultipleReadWrite() {
192         /**
193          * Creating multiple buffers.
194          */
195         ByteBuffer[] readBuffers = new ByteBuffer[10];
196         ByteBuffer[] writeBuffers = new ByteBuffer[10];
197 
198         for (int i = 0; i < writeBuffers.length; i++) {
199             readBuffers[i] = ByteBuffer.allocate(BUFFER_SIZE);
200             writeBuffers[i] = ByteBuffer.allocate(BUFFER_SIZE);
201 
202             randomGenerator = new Random(System.currentTimeMillis());
203 
204             randomGenerator.nextBytes(writeBuffers[i].array());
205         }
206 
207         for (int i = 0; i < writeBuffers.length; i++) {
208             Future<Void> returnVal1;
209             try {
210                 returnVal1 =
211                     session.write(writeBuffers[i], LOGICAL_BLOCK_ADDRESS, writeBuffers[i].remaining());
212 
213                 returnVal1.get();
214 
215                 assertTrue(!writeBuffers[i].equals(readBuffers[i]));
216             } catch (TaskExecutionException e) {
217                 Assert.fail();
218             } catch (InterruptedException e) {
219                 Assert.fail();
220             } catch (ExecutionException e) {
221                 Assert.fail();
222             }
223         }
224 
225         for (int i = 0; i < writeBuffers.length; i++) {
226             Future<Void> returnVal1;
227             try {
228                 returnVal1 = session.read(readBuffers[i], LOGICAL_BLOCK_ADDRESS, readBuffers[i].remaining());
229 
230                 returnVal1.get();
231 
232             } catch (TaskExecutionException e) {
233                 Assert.fail();
234             } catch (InterruptedException e) {
235                 Assert.fail();
236             } catch (ExecutionException e) {
237                 Assert.fail();
238             }
239         }
240 
241         for (int i = 0; i < writeBuffers.length; i++) {
242             assertTrue(writeBuffers[i].equals(readBuffers[writeBuffers.length - i - 1]));
243 
244             // Need to write them into a separate buffer since flip doesn't seem
245             // to work in a Array of ByteBuffers..
246             ByteBuffer write = ByteBuffer.allocate(BUFFER_SIZE);
247             ByteBuffer read = ByteBuffer.allocate(BUFFER_SIZE);
248 
249             write.put(writeBuffers[i]);
250             read.put(readBuffers[writeBuffers.length - i - 1]);
251 
252             write.flip();
253             read.flip();
254             assertTrue(write.equals(read));
255         }
