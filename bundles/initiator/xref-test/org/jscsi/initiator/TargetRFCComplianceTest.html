94           * @throws SAXException
95           * @throws ParserConfigurationException
96           * @throws IOException
97           * @throws ConfigurationException
98           */
99          public static void start() throws SAXException, ParserConfigurationException, IOException,
100             ConfigurationException {
101             if (isWindows()) {
102                 targetConfigurationFile = new File(CONFIG_DIR, "jscsi-target-windows.xml");
103             } else {
104                 targetConfigurationFile = new File(CONFIG_DIR, "jscsi-target-linux.xml");
105             }
106 
107             org.jscsi.target.Configuration targetConfiguration =
108                 org.jscsi.target.Configuration.create(new File(CONFIG_DIR, "jscsi-target.xsd"),
109                     targetConfigurationFile);
110 
111             target = new TargetServer(targetConfiguration);
112 
113             // Getting an Executor
114             threadPool = Executors.newSingleThreadExecutor();
115             // Starting the target
116             threadPool.submit(target);
117         }
118 
119         /**
120          * After the tests are finished shutdown the threadpool.
121          */
