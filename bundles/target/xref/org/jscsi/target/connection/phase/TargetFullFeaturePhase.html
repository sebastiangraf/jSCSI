33  /**
34   * Objects of this class represent the Target Full Feature Phase of a
35   * connection.
36   * 
37   * @see TargetPhase
38   * @author Andreas Ergenzinger
39   */
40  public final class TargetFullFeaturePhase extends TargetPhase {
41  
42      private static final Logger LOGGER = LoggerFactory.getLogger(TargetFullFeaturePhase.class);
43  
44      /**
45       * The current stage of this phase.
46       */
47      private TargetFullFeatureStage stage;
48  
49      /**
50       * While this variable is <code>true</code> the phase is still running,
51       * either executing a specific stage or waiting for the next one to begin.
52       */
53      private boolean running;
54  
55      /**
56       * The constructor.
57       * 
58       * @param connection
59       *            {@inheritDoc}
60       */
61      public TargetFullFeaturePhase(Connection connection) {
62          super(connection);
63      }
64  
65      /**
66       * Starts the full feature phase.
67       * 
68       * @return {@inheritDoc}
69       * @throws OperationNotSupportedException
70       *             {@inheritDoc}
71       * @throws IOException
72       *             {@inheritDoc}
73       * @throws InterruptedException
74       *             {@inheritDoc}
75       * @throws InternetSCSIException
76       *             {@inheritDoc}
77       * @throws DigestException
78       *             {@inheritDoc}
79       * @throws SettingsException
80       *             {@inheritDoc}
81       */
82      public boolean execute() throws DigestException, IOException, InterruptedException,
83          InternetSCSIException, SettingsException {
84  
85          running = true;
86          while (running) {
87  
88              ProtocolDataUnit pdu = connection.receivePdu();
89              BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
90  
91              // identify desired stage
92              switch (bhs.getOpCode()) {
93  
94              case SCSI_COMMAND:
95                  if (connection.getTargetSession().isNormalSession()) {
96                      final SCSICommandParser parser = (SCSICommandParser)bhs.getParser();
97                      ScsiOperationCode scsiOpCode = ScsiOperationCode.valueOf(parser.getCDB().get(0));
98  
99                      LOGGER.debug("scsiOpCode = " + scsiOpCode);// log SCSI
100                                                                // Operation Code
101 
102                     if (scsiOpCode != null) {
103                         switch (scsiOpCode) {
104                         case TEST_UNIT_READY:
105                             stage = new TestUnitReadyStage(this);
106                             break;
107                         case REQUEST_SENSE:
108                             stage = new RequestSenseStage(this);
109                             break;
110                         case FORMAT_UNIT:
111                             stage = new FormatUnitStage(this);
112                             break;
113                         case INQUIRY:
114                             stage = new InquiryStage(this);
115                             break;
116                         case MODE_SELECT_6:
117                             stage = null;
118                             scsiOpCode = null;
119                             break;
120                         case MODE_SENSE_6:
121                             stage = new ModeSenseStage(this);
122                             if (!((ModeSenseStage)stage).canHandle(pdu)) {
123                                 stage = null;
124                                 scsiOpCode = null;
125                             }
126                             break;
127                         case SEND_DIAGNOSTIC:
128                             stage = new SendDiagnosticStage(this);
129                             break;
130                         case READ_CAPACITY_10:// use common read capacity stage
131                         case READ_CAPACITY_16:
132                             stage = new ReadCapacityStage(this);
133                             break;
134                         case WRITE_6:// use common write stage
135                         case WRITE_10:
136                             stage = new WriteStage(this);
137                             break;
138                         case READ_6:// use common read stage
139                         case READ_10:
140                             stage = new ReadStage(this);
141                             break;
142                         case REPORT_LUNS:
143                             stage = new ReportLunsStage(this);
144                             break;
145                         default:
146                             scsiOpCode = null;
147 
148                         }
149                     }// else, or if default block was entered (programmer error)
150                     if (scsiOpCode == null) {
151                         LOGGER.error("Unsupported SCSI OpCode 0x"
152                             + Integer.toHexString(parser.getCDB().get(0) & 255) + " in SCSI Command PDU.");
153                         stage = new UnsupportedOpCodeStage(this);
154                     }
155 
156                 } else {// session is discovery session
157                     throw new InternetSCSIException("received SCSI command in discovery session");
158                 }
159                 break; // SCSI_COMMAND
160 
161             case NOP_OUT:
162                 stage = new PingStage(this);
163                 break;
164             case TEXT_REQUEST:
165                 stage = new TextNegotiationStage(this);
166                 break;
167             case LOGOUT_REQUEST:
168                 stage = new LogoutStage(this);
169                 running = false;
170                 break;
171             default:
172                 throw new InternetSCSIException();
173             }
174 
175             // process the PDU
176             stage.execute(pdu);
177         }
178         return false;
179     }
