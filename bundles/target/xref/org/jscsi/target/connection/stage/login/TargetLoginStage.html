22  /**
23   * This class is an abstract super-class for stages of the {@link TargetLoginPhase} (see
24   * {@link TargetConnection} for a description of
25   * the relationship between sessions, connections, phases and sessions), namely
26   * the {@link LoginOperationalParameterNegotiationStage} and the {@link SecurityNegotiationStage}.
27   * <p>
28   * The stage is started by calling the {@link #execute(ProtocolDataUnit)} method with the first
29   * {@link ProtocolDataUnit} to be processed as part of the stage.
30   * <p>
31   * Of equal importance is the {@link #getNextStageNumber()} method, which must be used to find out which stage
32   * or phase will follow this one.
33   * 
34   * @author Andreas Ergenzinger
35   */

90      /**
91       * Returns <code>true</code>, if and only if the specified PDU is a Login
92       * Request PDU and the CSN and InitiatorTaskTag fields check out.
93       * 
94       * @param pdu
95       *            the PDU to check
96       * @return <code>true</code> if the PDU checks out
97       */
98      protected boolean checkPdu(ProtocolDataUnit pdu) {
99          final BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
100         final LoginRequestParser parser = (LoginRequestParser)bhs.getParser();
101         if (bhs.getOpCode() == OperationCode.LOGIN_REQUEST && parser.getCurrentStageNumber() == stageNumber
102             && bhs.getInitiatorTaskTag() == initiatorTaskTag)
103             return true;
104         return false;
105     }

125     /**
126      * Receives a sequence of Login Request PDUs (as indicated by the
127      * {@link LoginRequestParser#isContinueFlag()} and returns the concatenated
128      * content of the text data segments.
129      * 
130      * @param pdu
131      *            the first PDU of the sequence
132      * @return the concatenated content of the text data segments
133      * @throws InternetSCSIException
134      * @throws InterruptedException
135      * @throws IOException
136      * @throws DigestException
137      * @throws SettingsException
138      */
139     protected final String receivePduSequence(ProtocolDataUnit pdu) throws InternetSCSIException,
140         InterruptedException, IOException, DigestException, SettingsException {
141 
142         // StringBuilder for the key-value pairs received during this sequence
143         final StringBuilder stringBuilder = new StringBuilder();
144 
145         // for accessing the fields of the last received PDU
146         BasicHeaderSegment bhs;
147         LoginRequestParser parser;
148 
149         // begin sequence
150         int sequenceLength = 1;
151         while (sequenceLength <= session.getTargetServer().getConfig().getInMaxRecvTextPduSequenceLength()) {
152             bhs = pdu.getBasicHeaderSegment();
153             parser = (LoginRequestParser)bhs.getParser();
154 
155             // check PDU
156             if (!checkPdu(pdu)) {
157                 // send login reject and leave stage
158                 sendRejectPdu(LoginStatus.INVALID_DURING_LOGIN);
159                 throw new InternetSCSIException("Wrong PDU in TargetLoginStage");
160             }
161 
162             // PDU is okay, so append text data segment to stringBuilder
163             ReadWrite.appendTextDataSegmentToStringBuffer(pdu.getDataSegment(), stringBuilder);
164 
165             // remember what stage the initiator wants to transition to
166             requestedNextStageNumber = parser.getNextStageNumber();
167 
168             // continue?
169             if (parser.isContinueFlag()) {
170                 // send reception confirmation
171                 pdu = TargetPduFactory.createLoginResponsePdu(false,// transitFlag
172                     false,// continueFlag
173                     stageNumber,// currentStage
174                     stageNumber,// nextStage
175                     session.getInitiatorSessionID(),// initiatorSessionID
176                     session.getTargetSessionIdentifyingHandle(),// targetSessionIdentifyingHandle
177                     initiatorTaskTag, LoginStatus.SUCCESS,// status
178                     ByteBuffer.allocate(0));// dataSegment
179                 connection.sendPdu(pdu);
180 
181                 // receive the next pdu
182                 pdu = connection.receivePdu();
183             } else
184                 // sequence is over
185                 return stringBuilder.toString();
186         }
187 
188         // initiator's text PDU sequence was too long
189         // send login reject and leave stage
190         sendRejectPdu(LoginStatus.OUT_OF_RESOURCES);
191         throw new InternetSCSIException("Wrong PDU in TargetLoginStage");
192     }

194     /**
195      * Sends a Login Response PDU sequence containing the specified
196      * <i>key-value</i> pairs.
197      * 
198      * @param keyValuePairs
199      *            contains <i>key-value</i> pairs to send
200      * @param nextStage
201      *            indicates if the target is willing to transition to a
202      *            different stage
203      * @throws SettingsException
204      * @throws InterruptedException
205      * @throws IOException
206      * @throws InternetSCSIException
207      * @throws DigestException
208      */
209     protected final void sendPduSequence(final String keyValuePairs, final LoginStage nextStage)
210         throws SettingsException, InterruptedException, IOException, InternetSCSIException, DigestException {
211 
212         // some variables
213         ProtocolDataUnit pdu;
214         BasicHeaderSegment bhs;
215         LoginRequestParser parser;
216         boolean continueFlag = true;
217         boolean transitFlag = false;
218 
219         // split input string into text data segments
220         final ByteBuffer[] dataSegments = ReadWrite.stringToTextDataSegments(keyValuePairs,// string
221             settings.getMaxRecvDataSegmentLength());// bufferSize
222 
223         // send all data segments (and receive confirmations)
224         for (int i = 0; i < dataSegments.length; ++i) {
225 
226             // adjust flags
227             if (i == dataSegments.length - 1) {
228                 continueFlag = false;
229                 if (stageNumber != nextStage)
230                     transitFlag = true;
231             }
232 
233             // create and send PDU
234             pdu = TargetPduFactory.createLoginResponsePdu(transitFlag,// transitFlag
235                 continueFlag,// continueFlag
236                 stageNumber,// currentStage
237                 nextStage,// nextStage
238                 session.getInitiatorSessionID(),// initiatorSessionID
239                 session.getTargetSessionIdentifyingHandle(),// targetSessionIdentifyingHandle
240                 initiatorTaskTag, LoginStatus.SUCCESS,// status
241                 dataSegments[i]);// dataSegment
242             connection.sendPdu(pdu);
243 
244             // receive confirmation
245             if (continueFlag) {
246                 // receive and check
247                 pdu = connection.receivePdu();
248                 bhs = pdu.getBasicHeaderSegment();
249                 parser = (LoginRequestParser)bhs.getParser();
250                 if (!checkPdu(pdu) || parser.isContinueFlag()) {
251                     // send login reject and leave stage
252                     sendRejectPdu(LoginStatus.INITIATOR_ERROR);
253                     throw new InternetSCSIException();
254                 }
255             }
256         }
257     }
