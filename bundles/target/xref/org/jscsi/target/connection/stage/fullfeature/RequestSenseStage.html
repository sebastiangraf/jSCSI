30   * <p>
31   * The <code>REQUEST SENSE</code> command requests that the device server transfer {@link SenseData} to the
32   * application client.
33   * 
34   * @author Andreas Ergenzinger
35   */
36  public class RequestSenseStage extends TargetFullFeatureStage {

39          super(targetFullFeaturePhase);
40      }

42      @Override
43      public void execute(ProtocolDataUnit pdu) throws IOException, InterruptedException,
44          InternetSCSIException, DigestException, SettingsException {
45  
46          final BasicHeaderSegment bhs = pdu.getBasicHeaderSegment();
47          final SCSICommandParser parser = (SCSICommandParser)bhs.getParser();
48  
49          ProtocolDataUnit responsePDU = null;// the response PDU

51          // get command details in CDB
52          final RequestSenseCdb cdb = new RequestSenseCdb(parser.getCDB());
53          final FieldPointerSenseKeySpecificData[] illegalFieldPointers = cdb.getIllegalFieldPointers();

55          if (illegalFieldPointers != null) {
56              // an illegal request has been made
57

64                      SenseKey.ILLEGAL_REQUEST,// sense key
65                      AdditionalSenseCodeAndQualifier.INVALID_FIELD_IN_CDB,// additional
66                                                                           // sense
67                                                                           // code
68                                                                           // and
69                                                                           // qualifier
70                      new SenseDataDescriptor[0]);// sense data descriptors
71  
72              } else {
73                  // fixed format sense data has been requested
74  
75                  senseData = new FixedFormatSenseData(false,// valid
76                      ErrorType.CURRENT,// error type
77                      false,// file mark
78                      false,// end of medium
79                      false,// incorrect length indicator
80                      SenseKey.ILLEGAL_REQUEST,// sense key
81                      new FourByteInformation(),// information
82                      new FourByteInformation(),// command specific
83                                                // information
84                      AdditionalSenseCodeAndQualifier.INVALID_FIELD_IN_CDB,// additional
85                                                                           // sense
86                                                                           // code
87                                                                           // and
88                                                                           // qualifier
89                      (byte)0,// field replaceable unit code
90                      illegalFieldPointers[0],// sense key specific data, only
91                                              // report first problem
92                      new AdditionalSenseBytes());// additional sense bytes
93              }
94  
95              responsePDU = TargetPduFactory.createSCSIResponsePdu(false,// bidirectionalReadResidualOverflow
96                  false,// bidirectionalReadResidualUnderflow
97                  false,// residualOverflow
98                  false,// residualUnderflow,
99                  SCSIResponseParser.ServiceResponse.TARGET_FAILURE,// response,
100                 SCSIStatus.CHECK_CONDITION,// status,

102                 0,// snackTag
103                 0,// expectedDataSequenceNumber
104                 0,// bidirectionalReadResidualCount
105                 0,// residualCount

107                                                                                                 // segment

109         } else {
110             /*
111              * PDU is okay
112              * 
113              * carry out command
114              * 
115              * Sense data shall be available and cleared under the conditions
116              * defined in SAM-3. If the device server has no other sense data
117              * available to return, it shall return the sense key set to NO
118              * SENSE and the additional sense code set to NO ADDITIONAL SENSE
119              * INFORMATION. This will always be the case with the jSCSI Target.
120              */

123

126                 AdditionalSenseCodeAndQualifier.NO_ADDITIONAL_SENSE_INFORMATION;

149                     (byte)0,// field replaceable unit code
150                     null,// sense key specific data, only report first
151                          // problem

153             }
154

156                 false,// bidirectionalReadResidualUnderflow

159                 SCSIResponseParser.ServiceResponse.COMMAND_COMPLETED_AT_TARGET,// response,
160                 SCSIStatus.GOOD,// status,
161                 bhs.getInitiatorTaskTag(),// initiatorTaskTag,
162                 0,// snackTag
163                 0,// expectedDataSequenceNumber
164                 0,// bidirectionalReadResidualCount
165                 0,// residualCount
166                 new ScsiResponseDataSegment(senseData, parser.getExpectedDataTransferLength()));// data
167                                                                                                 // segment
168         }
