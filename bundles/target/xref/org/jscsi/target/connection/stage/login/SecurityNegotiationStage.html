160     /**
161      * Checks if a the parameter is the key of an AuthMethod Key, which means
162      * one of the following (where &#60;key&#62; depends on the AuthMethod
163      * prefix):
164      * <ul>
165      * <li>CHAP_&#60;key&#62;/il>
166      * <li>KRB_&#60;key&#62;/il>
167      * <li>SPKM_&#60;key&#62;/il>
168      * <li>SRP_&#60;key&#62;/il>
169      * </ul>
170      * 
171      * @param <i>key</i> part of a <i>key-value</i> pair
172      * @return <code>true</code> if the String is an AuthMethod key, <code>false</code> if it is not.
173      */
174     private final boolean isAuthenticationKey(final String key) {
175         if (key == null || key.length() < 5)
176             return false;
177         final String fourChars = key.substring(0, 4);
178         final String fiveChars = key.substring(0, 5);
179         if ("CHAP_".matches(fiveChars) || "KRB_".matches(fourChars) || "SPKM_".matches(fiveChars)
180             || "SRP_".matches(fourChars)
181             || (key.length() >= 10 && "TargetAuth".matches(key.substring(0, 10))))
182             return true;
183         return false;
184     }
