63       * all connection-specific {@link Entry} objects, since the constructor will try to locate a specific

66       * @param entries

68       */
69      ConnectionSettingsBuilderComponent(final Collection<Entry> entries) {
70          try {
71              targetName = SettingsNegotiator.getEntry(TextKeyword.TARGET_NAME, entries).getStringValue();
72              dataDigest = SettingsNegotiator.getEntry(TextKeyword.DATA_DIGEST, entries).getStringValue();
73              headerDigest = SettingsNegotiator.getEntry(TextKeyword.HEADER_DIGEST, entries).getStringValue();
74              ifMarker = SettingsNegotiator.getEntry(TextKeyword.IF_MARKER, entries).getBooleanValue();
75              ifMarkInt = SettingsNegotiator.getEntry(TextKeyword.IF_MARK_INT, entries).getIntegerValue();
76              maxRecvDataSegmentLength =
77                  SettingsNegotiator.getEntry(TextKeyword.MAX_RECV_DATA_SEGMENT_LENGTH, entries)
78                      .getIntegerValue();
79              ofMarker = SettingsNegotiator.getEntry(TextKeyword.OF_MARKER, entries).getBooleanValue();
80              ofMarkInt = SettingsNegotiator.getEntry(TextKeyword.OF_MARK_INT, entries).getIntegerValue();
81          } catch (OperationNotSupportedException e) {
82              e.printStackTrace();
83          }
