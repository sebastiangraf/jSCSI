18  /**
19   * In addition to offering all methods of their {@link SettingsNegotiator} parent class,
20   * {@link ConnectionSettingsNegotiator} instances provide all
21   * methods necessary for text parameter negotiation and for creating {@link Settings} objects, which allow
22   * access to the current parameters.
23   * <p>
24   * Each instance of this class belongs to one {@link TargetConnection}, for which it manages all
25   * connection-specific parameters. A similar association exists between the connection's enclosing
26   * {@link TargetSession} and the {@link #sessionSettingsNegotiator} member variable.
27   * <p>
28   * A negotiation sequence consists of the following steps, performed in the given order:
29   * <p>
30   * <ol>
31   * <li>Call {@link #beginNegotiation()} until it returns <code>true</code>.</li>
32   * <li>One or, if the initiator is using multiple PDU sequences, multiple calls of
33   * {@link #negotiate(TargetServer, LoginStage, boolean, boolean, Vector, Vector)}. The method will return
34   * <code>false</code> if there was a problem.</li>
35   * <li>Call {@link #checkConstraints()} to check more complex requirements. The method will return
36   * <code>false</code> if there was a problem.</li>
37   * <li>Call {@link #finishNegotiation(boolean)} with the appropriate parameter. This step is mandatory.</li>
38   * </ol>
39   * 
40   * @author Andreas Ergenzinger, University of Konstanz
41   */
42  public final class ConnectionSettingsNegotiator extends SettingsNegotiator {

76      /**
77       * This method must be called at the beginning of a parameter negotiation
78       * sequence.
79       * <p>
80       * This method will block until no other {@link ConnectionSettingsNegotiator} is negotiating and will
81       * return <code>true</code> if negotiations may proceed, or <code>false</code>, if the request for the
82       * permission to negotiate was denied.
83       * 
84       * @return <code>true</code> if and only if the caller is allowed to
85       *         negotiate parameters
86       * @see #finishNegotiation(boolean)
87       */
88      public boolean beginNegotiation() {
89          final boolean hasLock = sessionSettingsNegotiator.lock();
90          if (hasLock) {
91              // back up entries with connection and with session scope
92              backUpEntries();
93              sessionSettingsNegotiator.backUpEntries();
94          }
95          return hasLock;
96      }

98      /**
99       * This method must be called at the end of a negotiation sequence. If there
100      * was a problem during negotiations, all changes to the managed {@link Entry} objects must be discarded
101      * by setting the
102      * <i>commitChanges</i> parameter to <code>false</code>. If it is <code>true</code>, then the parameter
103      * changes will be incorporated into
104      * the updated {@link #settings}.
105      * <p>
106      * Calling this method should be ensured by a <code>try ... catch ...
107      * finally</code> block.
108      * 
109      * @param commitChanges
110      *            <code>true</code> if and only if the negotiated parameter
111      *            changes are to be committed
112      */
113     public void finishNegotiation(final boolean commitChanges) {
114         // commit or roll back connection-specific entries
115         commitOrRollBackChanges(commitChanges);
116         // commit or roll back session-wide entries
117         sessionSettingsNegotiator.commitOrRollBackChanges(commitChanges);
118         // update settings
119         updateSettingsBuilderComponent();
120         sessionSettingsNegotiator.updateSettingsBuilderComponent();
121         updateSettings();
122         // allow other Threads to negotiate
123         sessionSettingsNegotiator.unlock();
124     }

126     /**
127      * Processes one or more <i>key-value</i> pairs sent by the initiator,
128      * formulates response <i>key-value</i> pairs and changes the involved {@link Entry} instances
129      * accordingly.
130      * 
131      * @param loginStage
132      *            specifies the current stage or phase
133      * @param leadingConnection
134      *            <code>true</code> if and only if the connection is the first
135      *            connection of the enclosing session
136      * @param initialPdu
137      *            <code>true</code> if and only if the <i>key-value</i> pairs
138      *            were sent in the first {@link ProtocolDataUnit} received on
139      *            this connection
140      * @param requestKeyValuePairs
141      *            contains the <i>key-value</i> pairs from the initiator;
142      *            processed elements will be removed
143      * @param responseKeyValuePairs
144      *            will contain the <i>key-value</i> pairs from the jSCSI target
145      * @return <code>true</code> if everything went fine and <code>false</code> if there was an irreconcilable
146      *         problem
147      */
148     public boolean negotiate(TargetServer target, final LoginStage loginStage,
149         final boolean leadingConnection, final boolean initialPdu, final List<String> requestKeyValuePairs,
150         final List<String> responseKeyValuePairs) {
151 
152         // split up key=value pairs from requester
153         final List<String> keys = new Vector<String>();
154         final List<String> values = new Vector<String>();
155         for (String keyValuePair : requestKeyValuePairs) {
156             final String[] split = TextParameter.splitKeyValuePair(keyValuePair);
157             if (split == null) {
158                 System.err.println("malformatted key=value pair: " + keyValuePair);
159                 return false;
160             }
161             keys.add(split[0]);
162             values.add(split[1]);
163         }
164 
165         // get respective entry (declared by initiator, or negotiated) and
166         // let it process the key=value pair
167         Entry entry;
168         boolean everythingOkay = true;
169         while (keys.size() > 0) {
170             entry = getEntry(keys.get(0));
171             // respond to unknown keys
172             if (entry == null) {
173                 responseKeyValuePairs.add(TextParameter.toKeyValuePair(keys.get(0),
174                     TextKeyword.NOT_UNDERSTOOD));
175             } else {// appropriate entry was found
176                 // process key=value pair and remember if there is any trouble
177                 everythingOkay &=
178                     entry.negotiate(target, loginStage, leadingConnection, initialPdu, keys.get(0), values
179                         .get(0), responseKeyValuePairs);
180             }
181             // remove processed key and values
182             keys.remove(0);
183             values.remove(0);
184         }
185 
186         // check if initiator has sent all mandatory parameters
187         // and append target declarations
188         if (initialPdu) {
189             // initiator must provide the InitiatorName in the first Login PDU
190             // (InitiatorAlias is optional)
191             everythingOkay &= getEntry(TextKeyword.INITIATOR_NAME).checkAccepted();
192 
193             // in a normal session the initiator must declare TargetName
194             // the target must reply with TargetPortalGroupTag and should
195             // append TargetAlias
196             boolean normalSession =
197                 TextKeyword.NORMAL.equals(((StringEntry)getEntry(TextKeyword.SESSION_TYPE)).getStringValue());
198             if (normalSession) {
199                 // check if proposed TargetName is correct
200                 final StringEntry targetNameEntry = (StringEntry)getEntry(TextKeyword.TARGET_NAME);
201                 final String targetName = targetNameEntry.getStringValue();
202                 if (targetName == null || // not declared
203                     !target.isValidTargetName(targetName))// wrong name
204                     everythingOkay = false;
205 
206                 // send TargetAlias
207                 String targetAlias = null;
208                 if (everythingOkay)
209                     targetAlias = target.getTarget(targetName).getTargetAlias();// might
210                 // be
211                 // undefined
212                 if (targetAlias != null) {
213                     responseKeyValuePairs.add(TextParameter.toKeyValuePair(TextKeyword.TARGET_ALIAS,
214                         targetAlias));
215                 }
216 
217                 // send TargetPortalGroupTag
218                 responseKeyValuePairs.add(TextParameter.toKeyValuePair(TextKeyword.TARGET_PORTAL_GROUP_TAG,
219                     Integer.valueOf(target.getConfig().getTargetPortalGroupTag()).toString()));
220             }
221         }
222 
223         return everythingOkay;
224     }

236     /**
237      * Returns the {@link Entry} responsible for negotiating the specified
238      * <i>key</i> identifying either a session-wide or connection-specific
239      * parameter, or <code>null</code> if no such {@link Entry} can be found.
240      * 
241      * @param key
242      *            identifies an {@link Entry}
243      * @return the requested {@link Entry} or <code>null</code>
244      */
245     private Entry getEntry(final String key) {
246         Entry entry = null;
247         // check connection-only entries
248         entry = getEntry(key, entries);
249         if (entry == null)// keep looking in session-wide entries
250             entry = sessionSettingsNegotiator.getEntry(key);
251         return entry;
252     }

254     /**
255      * This method checks if any parameter changes have been committed since the
256      * last call and, if so, creates a new {@link Settings} object reflecting
257      * the current parameters.
258      * 
259      * @return The current parameters

261     public Settings getSettings() {
262         // check if settings are up to date
263         if (sessionSettingsNegotiator.getCurrentSettingsId() > settings.getSettingsId())
264             updateSettings();
265         return settings;

295          */
296         entries.add(new StringEntry(new KeySet(TextKeyword.HEADER_DIGEST),// keySet
297             NegotiationType.NEGOTIATED,// negotiationType
298             Use.LOPNS,// use
299             NegotiationStatus.DEFAULT,// negotiationStatus
300             new String[] {
301                 TextKeyword.NONE
302             },// supportedValues,
303             TextKeyword.NONE));// defaultValue
304 
305         /*
306          * Turns the target-to-initiator markers on or off.
307          */
308         entries.add(new BooleanEntry(new KeySet(TextKeyword.IF_MARKER),// keySet
309             Use.LOPNS,// use
310             NegotiationStatus.DEFAULT,// negotiationStatus
311             false,// negotiationValue
312             BooleanResultFunction.AND,// resultFunction
313             false));// defaultValue
314 
315         /*
316          * Interval value (in 4-byte words) for target-to-initiator markers. The
317          * interval is measured from the end of one marker to the beginning of
318          * the next one. The offer can have only a range; the response can have
319          * only a single value (picked from the offered range) or Reject.
320          * 
321          * Will always be Irrelevant.
322          */
323         entries.add(new NumericalRangeEntry(new KeySet(TextKeyword.IF_MARK_INT),// keySet
324             Use.LOPNS,// use
325             NegotiationStatus.IRRELEVANT,// negotiationStatus
326             2048,// negotiationValue
327             NumericalValueRange.create(1, 65535),// protocolValueRange
328             2048));// defaultValue
329 
330         /*
331          * The maximum amount of data that either the initiator or the target
332          * can receive in any iSCSI PDU. Zero (don't care) can be used. I&T can
333          * specify (declare) the Max they can receive. This is a connection- and
334          * direction- specific parameter. The actual value used by the target
335          * will be min(This value, MaxBurstLength) for data-in and solicited
336          * data-out data. Min(This value, FirstBurstLength) for unsolicited
337          * data.
338          */
339         entries.add(new NumericalEntry(new KeySet(TextKeyword.MAX_RECV_DATA_SEGMENT_LENGTH),// keySet
340             NegotiationType.DECLARED,// negotiationType
341             Use.LOPNS_AND_FFP,// use
342             NegotiationStatus.DEFAULT,// negotiationStatus
343             8192,// negotiationValue (default value, will be used if I sends
344                  // 0)
345             NumericalValueRange.create(512, 16777215),// protocolValueRange
346             // 512 to 2^24 - 1
347             NumericalResultFunction.MIN,// resultFunction
348             8192,// defaultValue, 8K
349             true));// zeroMeansDontCare
350 
351         /*
352          * Turns the initiator-to-target markers on or off.
353          */
354         entries.add(new BooleanEntry(new KeySet(TextKeyword.OF_MARKER),// keySet
355             Use.LOPNS,// use
356             NegotiationStatus.DEFAULT,// negotiationStatus
357             false,// negotiationValue
358             BooleanResultFunction.AND,// resultFunction
359             false));// defaultValue
360 
361         /*
362          * Interval value (in 4-byte words) for initiator-to-target markers. The
363          * interval is measured from the end of one marker to the beginning of
364          * the next one. The offer can have only a range; the response can have
365          * only a single value (picked from the offered range) or Reject.
366          * 
367          * Will always be Irrelevant.
368          */
369         entries.add(new NumericalRangeEntry(new KeySet(TextKeyword.OF_MARK_INT),// keySet
370             Use.LOPNS,// use
371             NegotiationStatus.IRRELEVANT,// negotiationStatus
372             2048,// negotiationValue
373             NumericalValueRange.create(1, 65535),// protocolValueRange
374             2048));// defaultValue

379          * the initiator has to declare at the beginning of normal sessions.

381         entries.add(new StringEntry(new KeySet(TextKeyword.TARGET_NAME),// keySet
382             NegotiationType.DECLARED,// negotiationType
383             Use.INITIAL,// use
384             NegotiationStatus.NOT_NEGOTIATED,// negotiationStatus
