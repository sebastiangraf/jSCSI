43  /**
44   * Base class of all parsers tests.
45   *

48  public abstract class ProtocolDataUnitTest {
49  
50      // --------------------------------------------------------------------------
51      // --------------------------------------------------------------------------
52  
53      /** The used Protocol Data Unit for this test. */
54      protected ProtocolDataUnit protocolDataUnit;
55  
56      /** The used parser for this Protocol Data Unit. */
57      protected AbstractMessageParser recognizedParser;
58  
59      /** Size in bytes of an chunk of this object. */
60      protected int chunkSize;

62      // --------------------------------------------------------------------------
63      // --------------------------------------------------------------------------
64  
65      /** Default constructor to create a new, empty ProtocolDataUnitTest object. */
66      protected ProtocolDataUnitTest() {
67  
68          protocolDataUnit = new ProtocolDataUnitFactory().create("None", "None");
69          chunkSize = 8192;

76       * @param immediateFlag
77       *            The expected value of the immediate flag.
78       * @param finalFlag
79       *            The expected value of the final flag.
80       * @param opCode
81       *            The expected value of the operation code.
82       * @param totalAHSLength
83       *            The expected value of the total AHS length.
84       * @param dataSegmentLength
85       *            The expected value of the data segment length.
86       * @param initiatorTaskTag
87       *            The expected value of the initiator task tag.
88       * @throws IOException
89       *             This exception should be never thrown.
90       * @throws InternetSCSIException
91       *             This exception should be never thrown.
92       */
93      protected void testDeserialize(final boolean immediateFlag, final boolean finalFlag,
94          final OperationCode opCode, final int totalAHSLength, final int dataSegmentLength,
95          final int initiatorTaskTag) throws IOException, InternetSCSIException {
96  
97          testBasicHeaderSegment(immediateFlag, finalFlag, opCode, totalAHSLength, dataSegmentLength,
98              initiatorTaskTag);
99      }

101     /**
102      * Tests a given <code>SettingsMap</code> object against the parsed <code>IDataSegment</code> object
103      * contained in <code>protocolDataUnit</code> .

106      *            <code>SettingsMap</code> instance, which contains all

112     }

114     /**
115      * Tests the given Ethereal log against the parsed <code>IDataSegment</code> object contained in
116      * <code>protocolDataUnit</code>.
117      * 
118      * @param etherealLog
119      *            A String, which contains an Ethereal log.
120      */
121     protected final void testDataSegment(final String etherealLog) {
122 
123         testDataSegment(WiresharkMessageParser.parseToByteBuffer(etherealLog));
124     }
125 
126     /**
127      * Tests the given Ethereal log against the parsed <code>IDataSegment</code> object contained in
128      * <code>protocolDataUnit</code>.
129      * 
130      * @param etherealLog
131      *            A <code>ByteBuffer</code>, which contains an Ethereal log.
132      */
133     protected final void testDataSegment(final ByteBuffer etherealLog) {
134 
135         IDataSegment dataSegment =
136             DataSegmentFactory.create(etherealLog, recognizedParser.getDataSegmentFormat(), chunkSize);
137 
138         testDataSegment(dataSegment);
139     }
140 
141     /**
142      * Tests the <code>IDataSegment</code> instance to the parsed data segment
143      * of the <code>protocolDataUnit</code> instance.
144      * 
145      * @param expectedDataSegment
146      *            Uses this data segment for comparsion.
147      * 
148      */
149     protected final void testDataSegment(final IDataSegment expectedDataSegment) {
150 
151         IDataSegment testDataSegment =
152             DataSegmentFactory.create(protocolDataUnit.getDataSegment(), recognizedParser
153                 .getDataSegmentFormat(), chunkSize);
154         assertTrue(expectedDataSegment.equals(testDataSegment));
155     }

162      * the given file content.

165      *            The content of the test case to read from.

169      *             This exception should be never thrown.
170      * @throws DigestException
171      *             This exception should be never thrown.
172      */
173     protected void setUp(final String fileContent) throws InternetSCSIException, IOException, DigestException {
174 
175         if (protocolDataUnit != null) {
176             protocolDataUnit.deserialize(WiresharkMessageParser.parseToByteBuffer(fileContent));
177         }
178     }

180     /**
181      * This method resets the settings of this ProtocolDataUnit object.
182      */
183     @After
184     public void tearDown() {

186         protocolDataUnit.clear();
187         chunkSize = 8192;
188         recognizedParser = null;
189     }

194     /**
195      * Tests the fields, which are common in all parsers.
196      * 
197      * @param immediateFlag
198      *            The expected value of the immediate flag.
199      * @param finalFlag
200      *            The expected value of the final flag.
201      * @param opCode
202      *            The expected value of the operation code.
203      * @param totalAHSLength
204      *            The expected value of the total AHS length.
205      * @param dataSegmentLength
206      *            The expected value of the data segment length.
207      * @param initiatorTaskTag
208      *            The expected value of the initiator task tag.
209      */
210     protected void testBasicHeaderSegment(final boolean immediateFlag, final boolean finalFlag,
211         final OperationCode opCode, final int totalAHSLength, final int dataSegmentLength,
212         final int initiatorTaskTag) {
213 
214         BasicHeaderSegment bhs = protocolDataUnit.getBasicHeaderSegment();
215 
216         // test BHS fields
217         assertEquals(immediateFlag, bhs.isImmediateFlag());
218         assertEquals(finalFlag, bhs.isFinalFlag());
219         assertEquals(opCode, bhs.getOpCode());
220         assertEquals((byte)totalAHSLength, bhs.getTotalAHSLength());
221         assertEquals(dataSegmentLength, bhs.getDataSegmentLength());
222         assertEquals(initiatorTaskTag, bhs.getInitiatorTaskTag());
223 
224         recognizedParser = bhs.getParser();
225     }
