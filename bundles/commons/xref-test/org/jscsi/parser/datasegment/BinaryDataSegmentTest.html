61      @Test
62      public final void testDeserializeBuffer() {
63  
64          final BinaryDataSegment dataSegment = new BinaryDataSegment(CHUNK_SIZE);
65          assertNotNull(dataSegment);
66          assertEquals(0, dataSegment.getLength());
67  
68          final ByteBuffer testBuffer = ByteBuffer.wrap(TEST_CASE_1_ARRAY);
69          dataSegment.deserialize(testBuffer, TEST_CASE_1_ARRAY.length);
70          assertEquals(TEST_CASE_1_ARRAY.length, dataSegment.getLength());
71  
72          final ByteBuffer expectedResult = ByteBuffer.wrap(TEST_CASE_1_ARRAY_LONG);
73          dataSegment.dataBuffer.rewind();
74          assertTrue(expectedResult.equals(dataSegment.dataBuffer));
75      }

94      @Test
95      public final void testSerializeBuffer() {
96  
97          final BinaryDataSegment dataSegment = new BinaryDataSegment(CHUNK_SIZE);
98          assertNotNull(dataSegment);
99          assertEquals(0, dataSegment.getLength());
100 
101         final ByteBuffer testBuffer = ByteBuffer.wrap(TEST_CASE_1_ARRAY);
102         dataSegment.deserialize(testBuffer, TEST_CASE_1_ARRAY.length);
103         assertEquals(TEST_CASE_1_ARRAY.length, dataSegment.getLength());
104 
105         final ByteBuffer expectedResult = ByteBuffer.wrap(TEST_CASE_1_ARRAY_LONG);
106         dataSegment.dataBuffer.rewind();
107 
108         final ByteBuffer exportedDataSegment =
109             ByteBuffer.allocate(AbstractDataSegment.getTotalLength(dataSegment.getLength()));
110         dataSegment.serialize(exportedDataSegment, 0);
111         exportedDataSegment.rewind();
112 
113         assertTrue(expectedResult.equals(exportedDataSegment));

151         assertEquals(5, dataSegment.getLength());

153         final BinaryDataSegment anotherDataSegment = new BinaryDataSegment(CHUNK_SIZE);
154         testBuffer.rewind();

156         assertNotNull(anotherDataSegment);

175     /**
176      * Tests the correct functionality of the serialize and the deserialize
177      * method.
178      */
179     @Test
180     public final void testSerializeAndDeserialize() {
181 
182         final ByteBuffer testBuffer = ByteBuffer.wrap(TEST_CASE_1_ARRAY);
183         final BinaryDataSegment dataSegment = new BinaryDataSegment(CHUNK_SIZE);
184         assertEquals(0, dataSegment.getLength());
185 
186         dataSegment.append(testBuffer, testBuffer.remaining());
187         assertNotNull(dataSegment);
188         assertEquals(TEST_CASE_1_ARRAY.length, dataSegment.getLength());
189 
190         final int totalSize = AbstractDataSegment.getTotalLength(TEST_CASE_1_ARRAY.length);
191         final ByteBuffer serialized = ByteBuffer.allocate(totalSize);
192         assertEquals(totalSize, dataSegment.serialize(serialized, 0));
193         serialized.rewind();
194 
195         final BinaryDataSegment anotherDataSegment = new BinaryDataSegment(CHUNK_SIZE);
196         assertNotNull(anotherDataSegment);
197         assertEquals(0, anotherDataSegment.getLength());
198 
199         assertEquals(totalSize, anotherDataSegment.deserialize(serialized, serialized.remaining()));
200         assertNotNull(anotherDataSegment);
201         assertEquals(totalSize, anotherDataSegment.getLength());
202     }
