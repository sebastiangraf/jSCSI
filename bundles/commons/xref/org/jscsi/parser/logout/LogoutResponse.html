32  /**
33   * <h1>LogoutResponse</h1>
34   * <p>
35   * This enumeration defines all the Logout Responses, which the iSCSI Standard (RFC3720) defines.
36   * 
37   * @author Volker Wildi
38   */
39  public enum LogoutResponse {
40  
41      /**
42       * Connection or session closed successfully.
43       */
44      CONNECTION_CLOSED_SUCCESSFULLY((byte)0),
45  
46      /**
47       * CID not found.
48       */
49      CID_NOT_FOUND((byte)1),
50  
51      /**
52       * Connection recovery is not supported. If Logout reason code was recovery
53       * and target does not support it as indicated by the ErrorRecoveryLevel.
54       */
55      CONNECTION_RECOVERY_NOT_SUPPORTED((byte)2),
56  
57      /**
58       * Cleanup failed for various reasons.
59       */
60      CLEANUP_FAILED((byte)3);
61  
62      private final byte value;
63  
64      private static Map<Byte, LogoutResponse> mapping;
65  
66      static {
67          LogoutResponse.mapping = new HashMap<Byte, LogoutResponse>();
68          for (LogoutResponse s : values()) {
69              LogoutResponse.mapping.put(s.value, s);
70          }
71      }
72  
73      private LogoutResponse(final byte newValue) {
74  
75          value = newValue;
76      }
77  
78      /**
79       * Returns the value of this enumeration.
80       * 
81       * @return The value of this enumeration.
82       */
83      public final byte value() {
84  
85          return value;
86      }
87  
88      /**
89       * Returns the constant defined for the given <code>value</code>.
90       * 
91       * @param value
92       *            The value to search for.
93       * @return The constant defined for the given <code>value</code>. Or <code>null</code>, if this value is
94       *         not defined by this
95       *         enumeration.
96       */
97      public static final LogoutResponse valueOf(final byte value) {
98  
99          return LogoutResponse.mapping.get(value);
100     }
