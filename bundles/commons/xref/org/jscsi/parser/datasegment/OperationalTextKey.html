36  public enum OperationalTextKey {
37      /**
38       * Use: During Login - Security Negotiation Senders: Initiator and Target
39       * Scope: connection AuthMethod = &lt;list-of-values&gt; The main item of
40       * security negotiation is the authentication method (AuthMethod). The
41       * authentication methods that can be used (appear in the list-of-values)
42       * are either those listed in the following table or are vendor-unique
43       * methods:
44       * <p/>
45       * <table border="1">
46       * <tr>
47       * <th>Name</th>
48       * <th>Description</th>
49       * </tr>
50       * <tr>
51       * <td>KRB5</td>
52       * <td>Kerberos V5 - defined in [RFC1510]</td>
53       * </tr>
54       * <tr>
55       * <td>SPKM1</td>
56       * <td>Simple Public-Key GSS-API Mechanism defined in [RFC2025]</td>
57       * </tr>
58       * <tr>
59       * <td>SPKM2</td>
60       * <td>Simple Public-Key GSS-API Mechanism defined in [RFC2025]</td>
61       * </tr>
62       * <tr>
63       * <td>SRP</td>
64       * <td>Secure Remote Password defined in [RFC2945]</td>
65       * </tr>
66       * <tr>
67       * <td>CHAP</td>
68       * <td>Challenge Handshake Authentication Protocol defined in [RFC1994]</td>
69       * </tr>
70       * <tr>
71       * <td>None</td>
72       * <td>No authentication</td>
73       * </tr>
74       * </table>
75       * <br/>
76       * The AuthMethod selection is followed by an "authentication exchange"
77       * specific to the authentication method selected. The authentication method
78       * proposal may be made by either the initiator or the target. However the
79       * initiator MUST make the first step specific to the selected
80       * authentication method as soon as it is selected. It follows that if the
81       * target makes the authentication method proposal the initiator sends the
82       * first keys(s) of the exchange together with its authentication method
83       * selection. The authentication exchange authenticates the initiator to the
84       * target, and optionally, the target to the initiator. Authentication is
85       * OPTIONAL to use but MUST be supported by the target and initiator. The
86       * initiator and target MUST implement CHAP. All other authentication
87       * methods are OPTIONAL. Private or public extension algorithms MAY also be
88       * negotiated for authentication methods. Whenever a private or public
89       * extension algorithm is part of the default offer (the offer made in
90       * absence of explicit administrative action) the implementer MUST ensure
91       * that CHAP is listed as an alternative in the default offer and "None" is
92       * not part of the default offer. Extension authentication methods MUST be
93       * named using one of the following two formats:
94       * <ol>
95       * <li>Z-reversed.vendor.dns_name.do_something=</li>
96       * <li>Z&lt;#&gt;&lt;IANA-registered-string&gt;=</li>
97       * </ol>
98       * Authentication methods named using the Z- format are used as private
99       * extensions. Authentication methods named using the Z# format are used as
100      * public extensions that must be registered with IANA and MUST be described
101      * by an informational RFC. For all of the public or private extension
102      * authentication methods, the method specific keys MUST conform to the
103      * format specified in Section 5.1 Text Format for standard-label. To
104      * identify the vendor for private extension authentication methods, we
105      * suggest you use the reversed DNS-name as a prefix to the proper digest
106      * names. The part of digest-name following Z- and Z# MUST conform to the
107      * format for standard-label specified in Section 5.1 Text Format. Support
108      * for public or private extension authentication methods is OPTIONAL. The
109      * following subsections define the specific exchanges for each of the
110      * standardized authentication methods. As mentioned earlier the first step
111      * is always done by the initiator.
112      */
113     AUTH_METHOD("AuthMethod"),

115     /**
116      * Use: IO <br/>
117      * Senders: Initiator and Target<br/>
118      * Scope: CO
119      * <p/>
120      * HeaderDigest = &lt;list-of-values&gt; <br/>
121      * DataDigest = &lt;list-of-values&gt;
122      * <p/>
123      * Default is None for both HeaderDigest and DataDigest.
124      * <p/>
125      * Digests enable the checking of end-to-end, non-cryptographic data
126      * integrity beyond the integrity checks provided by the link layers and the
127      * covering of the whole communication path including all elements that may
128      * change the network level PDUs such as routers, switches, and proxies. <br/>
129      * The following table lists cyclic integrity checksums that can be
130      * negotiated for the digests and that MUST be implemented by every iSCSI
131      * initiator and target. These digest options only have error detection
132      * significance.
133      * <p/>
134      * <table border="1">
135      * <tr>
136      * <th>Name</th>
137      * <th>Description</th>
138      * <th>Generator</th>
139      * </tr>
140      * <tr>
141      * <td>CRC32CDigest</td>
142      * <td>32 bit CRC</td>
143      * <td>0x11edc6f41</td>
144      * </tr>
145      * <tr>
146      * <td>None</td>
147      * <td colspan="2">no digest</td>
148      * </tr>
149      * </table>
150      * <br/>
151      * The generator polynomial for this digest is given in hex-notation (e.g.,
152      * 0x3b stands for 0011 1011 and the polynomial is x**5+X**4+x**3+x+1).
153      * <p/>
154      * When the Initiator and Target agree on a digest, this digest MUST be used
155      * for every PDU in Full Feature Phase. <br/>
156      * Padding bytes, when present in a segment covered by a CRC, SHOULD be set
157      * to 0 and are included in the CRC. <br/>
158      * The CRC MUST be calculated by a method that produces the same results as
159      * the following process: <br/>
160      * <ul>
161      * <li>The PDU bits are considered as the coefficients of a polynomial M(x)
162      * of degree n-1; bit 7 of the lowest numbered byte is considered the most
163      * significant bit (x^n-1), followed by bit 6 of the lowest numbered byte
164      * through bit 0 of the highest numbered byte (x^0).</li>
165      * <li>The most significant 32 bits are complemented.</li>
166      * <li>The polynomial is multiplied by x^32 then divided by G(x). The
167      * generator polynomial produces a remainder R(x) of degree <= 31.</li>
168      * <li>The coefficients of R(x) are considered a 32 bit sequence.</li>
169      * <li>The bit sequence is complemented and the result is the CRC.</li>
170      * <li>The CRC bits are mapped into the digest word. The x^31 coefficient in
171      * bit 7 of the lowest numbered byte of the digest continuing through to the
172      * byte up to the x^24 coefficient in bit 0 of the lowest numbered byte,
173      * continuing with the x^23 coefficient in bit 7 of next byte through x^0 in
174      * bit 0 of the highest numbered byte.</li>
175      * <li>Computing the CRC over any segment (data or header) extended to
176      * include the CRC built using the generator 0x11edc6f41 will always get the
177      * value 0x1c2d19ed as its final remainder (R(x)). This value is given here
178      * in its polynomial form (i.e., not mapped as the digest word).</li>
179      * </ul>
180      * For a discussion about selection criteria for the CRC, see [RFC3385]. For
181      * a detailed analysis of the iSCSI polynomial, see [Castagnoli93]. <br/>
182      * Private or public extension algorithms MAY also be negotiated for
183      * digests. Whenever a private or public digest extension algorithm is part
184      * of the default offer (the offer made in absence of explicit
185      * administrative action) the implementer MUST ensure that CRC32CDigest is
186      * listed as an alternative in the default offer and "None" is not part of
187      * the default offer. <br/>
188      * Extension digest algorithms MUST be named using one of the following two
189      * formats: <br/>
190      * <ol>
191      * <li>Y-reversed.vendor.dns_name.do_something=</li>
192      * <li>Y&lt;#&gt;&lt;IANA-registered-string&gt;=</li>
193      * </ol>
194      * Digests named using the Y- format are used for private purposes
195      * (unregistered). Digests named using the Y# format (public extension) must
196      * be registered with IANA and MUST be described by an informational RFC. <br/>
197      * For private extension digests, to identify the vendor, we suggest you use
198      * the reversed DNS-name as a prefix to the proper digest names. <br/>
199      * The part of digest-name following Y- and Y# MUST conform to the format
200      * for standard-label specified in Section 5.1 Text Format. Support for
201      * public or private extension digests is OPTIONAL.
202      */
203     HEADER_DIGEST("HeaderDigest"),
