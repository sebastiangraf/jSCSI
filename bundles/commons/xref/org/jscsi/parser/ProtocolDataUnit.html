45  /**
46   * <h1>ProtocolDataUnit</h1>
47   * <p>
48   * This class encapsulates a Protocol Data Unit (PDU), which is defined in the
49   * iSCSI Standard (RFC 3720).
50   * 
51   * @author Volker Wildi
52   */
53  public final class ProtocolDataUnit {
54  
55      // --------------------------------------------------------------------------
56      // --------------------------------------------------------------------------
57  
58      /** The initial size of the Additional Header Segment. */
59      private static final int AHS_INITIAL_SIZE = 0;
60  
61      /** The Log interface. */
62      private static final Log LOGGER = LogFactory.getLog(ProtocolDataUnit.class);
63  
64      // --------------------------------------------------------------------------
65      // --------------------------------------------------------------------------
66  
67      /** The Basic Header Segment of this PDU. */
68      private final BasicHeaderSegment basicHeaderSegment;
69  
70      /** The Additional Header Segment 1...n (optional) of this PDU. */
71      private final AbstractList<AdditionalHeaderSegment> additionalHeaderSegments;
72  
73      /**
74       * The (optional) Data Segment contains PDU associated data. Its payload
75       * effective length is provided in the BHS field -
76       * <code>DataSegmentLength</code>. The Data Segment is also padded to
77       * multiple of a <code>4</code> byte words.
78       */
79      private ByteBuffer dataSegment;
80  
81      /**
82       * Optional header and data digests protect the integrity of the header and
83       * data, respectively. The digests, if present, are located, respectively,
84       * after the header and PDU-specific data, and cover respectively the header
85       * and the PDU data, each including the padding bytes, if any.
86       * <p>
87       * <b>The existence and type of digests are negotiated during the Login
88       * Phase. </b>
89       * <p>
90       * The separation of the header and data digests is useful in iSCSI routing
91       * applications, in which only the header changes when a message is
92       * forwarded. In this case, only the header digest should be recalculated.
93       * <p>
94       * Digests are not included in data or header length fields.
95       * <p>
96       * A zero-length Data Segment also implies a zero-length data-digest.
97       */
98  
99      /** Digest of the header of this PDU. */
100     private IDigest headerDigest;
101 
102     /** Digest of the data segment of this PDU. */
103     private IDigest dataDigest;

108     /**
109      * Default constructor, creates a new, empty ProtcolDataUnit object.
110      * 
111      * @param initHeaderDigest
112      *            The instance of the digest to use for the Basic Header Segment
113      *            protection.
114      * @param initDataDigest
115      *            The instance of the digest to use for the Data Segment
116      *            protection.
117      */
118     public ProtocolDataUnit(final IDigest initHeaderDigest,
119             final IDigest initDataDigest) {
120 
121         basicHeaderSegment = new BasicHeaderSegment();
122         headerDigest = initHeaderDigest;
123 
124         additionalHeaderSegments = new ArrayList<AdditionalHeaderSegment>(
125                 AHS_INITIAL_SIZE);
126 
127         dataSegment = ByteBuffer.allocate(0);
128         dataDigest = initDataDigest;
129     }

134     /**
135      * Serialize all informations of this PDU object to its byte representation.
136      * 
137      * @return The byte representation of this PDU.
138      * @throws InternetSCSIException
139      *             If any violation of the iSCSI-Standard emerge.
140      * @throws IOException
141      *             if an I/O error occurs.
142      */
143     public final ByteBuffer serialize() throws InternetSCSIException,
144             IOException {
145 
146         basicHeaderSegment.getParser().checkIntegrity();
147 
148         final ByteBuffer pdu = ByteBuffer.allocate(calcSize());
149 
150         int offset = 0;
151         offset += basicHeaderSegment.serialize(pdu, offset);
152 
153         if (LOGGER.isTraceEnabled()) {
154             LOGGER.trace("Serialized Basic Header Segment:\n" + toString());
155         }
156 
157         offset += serializeAdditionalHeaderSegments(pdu, offset);
158 
159         // write header digest
160         // TODO: Move CRC calculation in BasicHeaderSegment.serialize?
161         if (basicHeaderSegment.getParser().canHaveDigests()) {
162             offset += serializeDigest(pdu, headerDigest);
163         }
164 
165         // serialize data segment
166         offset += serializeDataSegment(pdu, offset);
167 
168         // write data segment digest
169         // TODO: Move CRC calculation in BasicHeaderSegment.serialize?
170         if (basicHeaderSegment.getParser().canHaveDigests()) {
171             offset += serializeDigest(pdu, dataDigest);
172         }
173 
174         return (ByteBuffer) pdu.rewind();
175     }

177     /**
178      * Deserializes (parses) a given byte representation of a PDU to an PDU
179      * object.
180      * 
181      * @param pdu
182      *            The byte representation of an PDU to parse.
183      * @return The number of bytes, which are serialized.
184      * @throws InternetSCSIException
185      *             If any violation of the iSCSI-Standard emerge.
186      * @throws IOException
187      *             if an I/O error occurs.
188      * @throws DigestException
189      *             There is a mismatch of the digest.
190      */
191     public final int deserialize(final ByteBuffer pdu)
192             throws InternetSCSIException, IOException, DigestException {
193 
194         int offset = deserializeBasicHeaderSegment(pdu);
195 
196         offset += deserializeAdditionalHeaderSegments(pdu, offset);
197 
198         offset += deserializeDataSegment(pdu, offset);
199 
200         basicHeaderSegment.getParser().checkIntegrity();
201 
202         return offset;
203     }

205     /**
206      * Deserializes a given array starting from offset <code>0</code> and store
207      * the informations in the BasicHeaderSegment object..
208      * 
209      * @param bhs
210      *            The array to read from.
211      * @throws InternetSCSIException
212      *             If any violation of the iSCSI-Standard emerge.
213      * @throws DigestException
214      *             There is a mismatch of the digest.
215      */
216     private final int deserializeBasicHeaderSegment(final ByteBuffer bhs)
217             throws InternetSCSIException, DigestException {
218 
219         int len = basicHeaderSegment.deserialize(this, bhs);
220 
221         // read header digest and validate
222         if (basicHeaderSegment.getParser().canHaveDigests()) {
223             len += deserializeDigest(bhs, bhs.position()
224                     - BasicHeaderSegment.BHS_FIXED_SIZE,
225                     BasicHeaderSegment.BHS_FIXED_SIZE, headerDigest);
226         }
227 
228         if (LOGGER.isTraceEnabled()) {
229             LOGGER.trace("Deserialized Basic Header Segment:\n" + toString());
230         }
231 
232         return len;
233     }

235     /**
236      * Deserializes a array (starting from offset <code>0</code>) and store the
237      * informations to the <code>AdditionalHeaderSegment</code> object.
238      * 
239      * @param pdu
240      *            The array to read from.
241      * @return The length of the read bytes.
242      * @throws InternetSCSIException
243      *             If any violation of the iSCSI-Standard emerge.
244      */
245     private final int deserializeAdditionalHeaderSegments(final ByteBuffer pdu)
246             throws InternetSCSIException {
247 
248         return deserializeAdditionalHeaderSegments(pdu, 0);
249     }

251     /**
252      * Deserializes a array (starting from the given offset) and store the
253      * informations to the <code>AdditionalHeaderSegment</code> object.
254      * 
255      * @param pdu
256      *            The <code>ByteBuffer</code> to read from.
257      * @param offset
258      *            The offset to start from.
259      * @return The length of the written bytes.
260      * @throws InternetSCSIException
261      *             If any violation of the iSCSI-Standard emerge.
262      */
263     private final int deserializeAdditionalHeaderSegments(final ByteBuffer pdu,
264             final int offset) throws InternetSCSIException {
265 
266         // parsing Additional Header Segment
267         int off = offset;
268         int ahsLength = basicHeaderSegment.getTotalAHSLength();
269         while (ahsLength != 0) {
270             final AdditionalHeaderSegment tmpAHS = new AdditionalHeaderSegment();
271             tmpAHS.deserialize(pdu, off);
272 
273             additionalHeaderSegments.add(tmpAHS);
274             ahsLength -= tmpAHS.getLength();
275 
276             off += tmpAHS.getSpecificField().position();
277         }
278 
279         return off - offset;
280     }

282     /**
283      * Serialize all the contained additional header segments to the destination
284      * array starting from the given offset.
285      * 
286      * @param dst
287      *            The destination array to write in.
288      * @param offset
289      *            The offset to start to write in <code>dst</code>.
290      * @return The written length.
291      * @throws InternetSCSIException
292      *             If any violation of the iSCSI-Standard emerge.
293      */
294     private final int serializeAdditionalHeaderSegments(final ByteBuffer dst,
295             final int offset) throws InternetSCSIException {
296 
297         int off = offset;
298         for (AdditionalHeaderSegment ahs : additionalHeaderSegments) {
299             off += ahs.serialize(dst, off);
300         }
301 
302         return off - offset;
303     }

305     /**
306      * Serializes the data segment (binary or key-value pairs) to a destination
307      * array, staring from offset to write.
308      * 
309      * @param dst
310      *            The array to write in.
311      * @param offset
312      *            The start offset to start from in <code>dst</code>.
313      * @return The written length.
314      * @throws InternetSCSIException
315      *             If any violation of the iSCSI-Standard emerge.
316      */
317     public final int serializeDataSegment(final ByteBuffer dst, final int offset)
318             throws InternetSCSIException {
319 
320         dataSegment.rewind();
321         dst.position(offset);
322         dst.put(dataSegment);
323 
324         return dataSegment.limit();
325     }

327     /**
328      * Deserializes a array (starting from the given offset) and store the
329      * informations to the Data Segment.
330      * 
331      * @param pdu
332      *            The array to read from.
333      * @param offset
334      *            The offset to start from.
335      * @return The length of the written bytes.
336      * @throws IOException
337      *             if an I/O error occurs.
338      * @throws InternetSCSIException
339      *             If any violation of the iSCSI-Standard emerge.
340      * @throws DigestException
341      *             There is a mismatch of the digest.
342      */
343     private final int deserializeDataSegment(final ByteBuffer pdu,
344             final int offset) throws IOException, InternetSCSIException,
345             DigestException {
346 
347         final int length = basicHeaderSegment.getDataSegmentLength();
348 
349         if (dataSegment == null || dataSegment.capacity() < length) {
350             dataSegment = ByteBuffer.allocate(AbstractDataSegment
351                     .getTotalLength(length));
352         }
353 
354         dataSegment.put(pdu);
355         dataSegment.flip();
356 
357         // read data segment digest and validate
358         if (basicHeaderSegment.getParser().canHaveDigests()) {
359             deserializeDigest(pdu, offset, length, dataDigest);
360         }
361 
362         if (dataSegment == null) {
363             return 0;
364         } else {
365             return dataSegment.limit();
366         }
367     }

372     /**
373      * Writes this <code>ProtocolDataUnit</code> object to the given
374      * <code>SocketChannel</code>.
375      * 
376      * @param sChannel
377      *            <code>SocketChannel</code> to write to.
378      * @return The number of bytes written, possibly zero.
379      * @throws InternetSCSIException
380      *             if any violation of the iSCSI-Standard emerge.
381      * @throws IOException
382      *             if an I/O error occurs.
383      */
384     public final int write(final SocketChannel sChannel)
385             throws InternetSCSIException, IOException {
386 
387         // print debug informations
388         if (LOGGER.isTraceEnabled()) {
389             LOGGER.trace(basicHeaderSegment.getParser().getShortInfo());
390         }
391 
392         final ByteBuffer src = serialize();
393         int length = 0;
394 
395         while (length < src.limit()) {
396             length += sChannel.write(src);
397         }
398 
399         return length;
400     }

402     /**
403      * Reads from the given <code>SocketChannel</code> all the neccassary bytes
404      * to fill this PDU.
405      * 
406      * @param sChannel
407      *            <code>SocketChannel</code> to read from.
408      * @return The number of bytes, possibly zero,or <code>-1</code> if the
409      *         channel has reached end-of-stream
410      * @throws IOException
411      *             if an I/O error occurs.
412      * @throws InternetSCSIException
413      *             if any violation of the iSCSI-Standard emerge.
414      * @throws DigestException
415      *             if a mismatch of the digest exists.
416      */
417     public final int read(final SocketChannel sChannel)
418             throws InternetSCSIException, IOException, DigestException {
419 
420         // read Basic Header Segment first to determine the total length of this
421         // Protocol Data Unit.
422         clear();
423 
424         final ByteBuffer bhs = ByteBuffer
425                 .allocate(BasicHeaderSegment.BHS_FIXED_SIZE);
426         int len = 0;
427         while (len < BasicHeaderSegment.BHS_FIXED_SIZE) {
428             int lens = sChannel.read(bhs);
429             if (lens == -1) {
430                 // The Channel was closed at the Target (e.g. the Target does
431                 // not support Multiple Connections)
432                 throw new ClosedChannelException();
433             }
434             len += lens;
435             LOGGER.trace("Receiving through SocketChannel: " + len
436                     + " of maximal " + BasicHeaderSegment.BHS_FIXED_SIZE);
437 
438         }
439         bhs.flip();
440 
441         deserializeBasicHeaderSegment(bhs);
442         // check for further reading
443         if (getBasicHeaderSegment().getTotalAHSLength() > 0) {
444             final ByteBuffer ahs = ByteBuffer.allocate(basicHeaderSegment
445                     .getTotalAHSLength());
446             int ahsLength = 0;
447             while (ahsLength < getBasicHeaderSegment().getTotalAHSLength()) {
448                 ahsLength += sChannel.read(ahs);
449             }
450             len += ahsLength;
451             ahs.flip();
452 
453             deserializeAdditionalHeaderSegments(ahs);
454         }
455         if (basicHeaderSegment.getDataSegmentLength() > 0) {
456             dataSegment = ByteBuffer.allocate(AbstractDataSegment
457                     .getTotalLength(basicHeaderSegment.getDataSegmentLength()));
458             int dataSegmentLength = 0;
459             while (dataSegmentLength < basicHeaderSegment
460                     .getDataSegmentLength()) {
461                 dataSegmentLength += sChannel.read(dataSegment);
462             }
463             len += dataSegmentLength;
464             dataSegment.flip();
465         }
466 
467         // print debug informations
468         if (LOGGER.isTraceEnabled()) {
469             LOGGER.trace(basicHeaderSegment.getParser().getShortInfo());
470         }
471 
472         return len;
473     }

615     /**
616      * Calculates the needed size (in bytes) of serializing this object.
617      * 
618      * @return The needed size to store this object.
619      */
620     private final int calcSize() {
621 
622         int size = BasicHeaderSegment.BHS_FIXED_SIZE;
623         size += basicHeaderSegment.getTotalAHSLength()
624                 * AdditionalHeaderSegment.AHS_FACTOR;
625 
626         // plus the sizes of the used digests
627         size += headerDigest.getSize();
628         size += dataDigest.getSize();
629 
630         size += AbstractDataSegment.getTotalLength(basicHeaderSegment
631                 .getDataSegmentLength());
632 
633         return size;
634     }
635 
636     private final int serializeDigest(final ByteBuffer pdu, final IDigest digest) {
637 
638         final int size = digest.getSize();
639         if (size > 0) {
640             digest.reset();
641             pdu.mark();
642             digest.update(pdu, 0, BasicHeaderSegment.BHS_FIXED_SIZE);
643             pdu.putInt((int) digest.getValue());
644             pdu.reset();
645         }
646 
647         return size;
648     }
649 
650     private final int deserializeDigest(final ByteBuffer pdu, final int offset,
651             final int length, final IDigest digest) throws DigestException {
652 
653         pdu.mark();
654         digest.update(pdu, offset, length);
655         digest.validate();
656         pdu.reset();
657 
658         return digest.getSize();
659     }
