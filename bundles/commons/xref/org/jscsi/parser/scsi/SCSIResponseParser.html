39  /**
40   * <h1>SCSIResponseParser</h1>
41   * <p>
42   * This class parses a SCSI Response message defined in the iSCSI Standard (RFC3720).
43   * <p>
44   * <h4>StatSN - Status Sequence Number</h4> StatSN is a Sequence Number that the target iSCSI layer generates
45   * per connection and that in turn, enables the initiator to acknowledge status reception. StatSN is
46   * incremented by <code>1</code> for every response/status sent on a connection except for responses sent as a
47   * result of a retry or SNACK. In the case of responses sent due to a retransmission request, the StatSN MUST
48   * be the same as the first time the PDU was sent unless the connection has since been restarted.
49   * <p>
50   * <h4>ExpCmdSN - Next Expected CmdSN from this Initiator</h4> ExpCmdSN is a Sequence Number that the target
51   * iSCSI returns to the initiator to acknowledge command reception. It is used to update a local variable with
52   * the same name. An ExpCmdSN equal to <code>MaxCmdSN + 1</code> indicates that the target cannot accept new
53   * commands.
54   * <p>
55   * <h4>MaxCmdSN - Maximum CmdSN from this Initiator</h4> MaxCmdSN is a Sequence Number that the target iSCSI
56   * returns to the initiator to indicate the maximum CmdSN the initiator can send. It is used to update a local
57   * variable with the same name. If MaxCmdSN is equal to <code>ExpCmdSN - 1</code>, this indicates to the
58   * initiator that the target cannot receive any additional commands. When MaxCmdSN changes at the target while
59   * the target has no pending PDUs to convey this information to the initiator, it MUST generate a NOP-IN to
60   * carry the new MaxCmdSN.
61   * <p/>
62   * iSCSI targets MUST support and enable autosense. If Status is CHECK CONDITION (<code>0x02</code>), then the
63   * Data Segment MUST contain sense data for the failed command. <br/>
64   * For some iSCSI responses, the response data segment MAY contain some response related information, (e.g.,
65   * for a target failure, it may contain a vendor specific detailed description of the failure).
66   * 
67   * @author Volker Wildi, University of Konstanz
68   */
69  public class SCSIResponseParser extends TargetMessageParser {

75       * This enumerations defines all valid service responses, which are defined
76       * in the iSCSI Standard (RFC 3720).
77       * <p>
78       * 0x80-0xff - Vendor specific
79       */
80      public static enum ServiceResponse {
81          /** Command completed at target. */
82          COMMAND_COMPLETED_AT_TARGET((byte)0x00),
83          /** Target Failure. */
84          TARGET_FAILURE((byte)0x01);
85  
86          private final byte value;
87  
88          private static Map<Byte, ServiceResponse> mapping;
89  
90          static {
91              ServiceResponse.mapping = new HashMap<Byte, ServiceResponse>();
92              for (ServiceResponse s : values()) {
93                  ServiceResponse.mapping.put(s.value, s);
94              }
95          }
96  
97          private ServiceResponse(final byte newValue) {
98  
99              value = newValue;
100         }

431     /** {@inheritDoc} */
432     @Override
433     protected final void deserializeBytes1to3(final int line) throws InternetSCSIException {
434 
435         Utils.isReserved(line & RESERVED_FLAGS_MASK);
436 
437         bidirectionalReadResidualOverflow = Utils.isBitSet(line & Constants.READ_RESIDUAL_OVERFLOW_FLAG_MASK);
438         bidirectionalReadResidualUnderflow =
439             Utils.isBitSet(line & Constants.READ_RESIDUAL_UNDERFLOW_FLAG_MASK);
440         residualOverflow = Utils.isBitSet(line & Constants.RESIDUAL_OVERFLOW_FLAG_MASK);
441         residualUnderflow = Utils.isBitSet(line & Constants.RESIDUAL_UNDERFLOW_FLAG_MASK);
442         response =
443             ServiceResponse.valueOf((byte)((line & Constants.THIRD_BYTE_MASK) >>> Constants.ONE_BYTE_SHIFT));
444         status = SCSIStatus.valueOf((byte)(line & Constants.FOURTH_BYTE_MASK));
445     }

478     /** {@inheritDoc} */
479     @Override
480     protected final void checkIntegrity() throws InternetSCSIException {
481 
482         String exceptionMessage;
483         do {
484             Utils.isReserved(logicalUnitNumber);
485 
486             if (response != ServiceResponse.COMMAND_COMPLETED_AT_TARGET) {
487                 if (bidirectionalReadResidualOverflow || bidirectionalReadResidualUnderflow
488                     || residualOverflow || residualUnderflow) {
489                     exceptionMessage =
490                         "Theses bits must to be 0, because the command is not completed at the target.";
491                     break;
492                 }
493 
494                 if (status != SCSIStatus.GOOD) {
495                     exceptionMessage =
496                         "Status Code is only valid, because the command is not completed at the target.";
497                     break;
498                 }
499             }
500 
501             if (bidirectionalReadResidualOverflow && bidirectionalReadResidualUnderflow) {
502                 exceptionMessage = "The 'o' and 'u' bits must be set mutal exclusion.";
503                 break;
504             }
505 
506             if (residualOverflow && residualUnderflow) {
507                 exceptionMessage = "The 'O' and 'U' bits must be set mutal exclusion.";
508                 break;
509             }
510 
511             if ((!residualOverflow && !residualUnderflow) && residualCount != 0) {
512                 exceptionMessage =
513                     "ResidualCount is only valid either the ResidualOverflow or ResidualUnderflow-Flag is set.";
514                 break;
515             }
516 
517             if ((!bidirectionalReadResidualOverflow && !bidirectionalReadResidualUnderflow)
518                 && bidirectionalReadResidualCount != 0) {
519                 exceptionMessage =
520                     "BidirectionalResidualCount is only valid either the "
521                         + "BidirectionalResidualOverflow or BidirectionalResidualUnderflow-Flag is set.";
522                 break;
523             }
524 
525             if (response != ServiceResponse.COMMAND_COMPLETED_AT_TARGET && expectedDataSequenceNumber != 0) {
526                 exceptionMessage =
527                     "The ExpectedDataSequenceNumber is not valid, because the command is not "
528                         + "completed at the target.";
529                 break;
530             }
531 
532             // message is checked correctly
533             return;
534         } while (false);
535 
536         throw new InternetSCSIException(exceptionMessage);
537     }

542     /** {@inheritDoc} */
543     @Override
544     protected final int serializeBytes1to3() {
545 
546         int line = status.value();
547         line |= response.value() << Constants.ONE_BYTE_SHIFT;
548 
549         if (residualUnderflow) {
550             line |= Constants.RESIDUAL_UNDERFLOW_FLAG_MASK;
551         }
552 
553         if (residualOverflow) {
554             line |= Constants.RESIDUAL_OVERFLOW_FLAG_MASK;
555         }
556 
557         if (bidirectionalReadResidualUnderflow) {
558             line |= Constants.READ_RESIDUAL_UNDERFLOW_FLAG_MASK;
559         }
560 
561         if (bidirectionalReadResidualOverflow) {
562             line |= Constants.READ_RESIDUAL_OVERFLOW_FLAG_MASK;
563         }
564 
565         return line;
566     }
