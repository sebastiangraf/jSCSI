37  /**
38   * <h1>LoginRequestParser</h1>
39   * <p>
40   * This class parses a Login Request message defined in the iSCSI Standard (RFC3720).
41   * <p>
42   * After establishing a TCP connection between an initiator and a target, the initiator MUST start a Login
43   * Phase to gain further access to the target’s resources.
44   * <p>
45   * The Login Phase (see Chapter 5) consists of a sequence of Login Requests and Responses that carry the same
46   * Initiator Task Tag.
47   * <p>
48   * <b>Login Requests are always considered as immediate.</b>
49   * <p/>
50   * The version number of the current draft is <code>0x00</code>. As such, all devices MUST carry version
51   * <code>0x00</code> for both Version-min and Version-max.
52   * <p>
53   * Here the final flag has the following meaning: If set to <code>1</code>, indicates that the initiator is
54   * ready to transit to the next stage.
55   * <p>
56   * If the <code>T</code> bit is set to <code>1</code> and <code>NSG</code> is <code>FullFeaturePhase</code>,
57   * then this also indicates that the initiator is ready for the Final Login Response (see Chapter 5).
58   * <p>
59   * <h4>T (Transit) Bit</h4> If set to <code>1</code>, indicates that the initiator is ready to transit to the
60   * next stage.
61   * <p>
62   * If the T bit is set to <code>1</code> and NSG is FullFeaturePhase, then this also indicates that the
63   * initiator is ready for the Final Login Response (see Chapter 5).
64   * <p>
65   * <h4>CmdSN</h4> CmdSN is either the initial command sequence number of a session (for the first Login
66   * Request of a session - the "leading" login), or the command sequence number in the command stream if the
67   * login is for a new connection in an existing session.<br/>
68   * Examples:<br/>
69   * <ul>
70   * <li>Login on a leading connection - if the leading login carries the CmdSN <code>123</code>, all other
71   * Login Requests in the same Login Phase carry the CmdSN <code>123</code> and the first non-immediate command
72   * in FullFeaturePhase also carries the CmdSN <code>123</code>.</li>
73   * <li>Login on other than a leading connection - if the current CmdSN at the time the first login on the
74   * connection is issued is <code>500</code>, then that PDU carries <code>CmdSN=500</code>. Subsequent Login
75   * Requests that are needed to complete this Login Phase may carry a CmdSN higher than <code>500</code> if
76   * non-immediate requests that were issued on other connections in the same session advance CmdSN.</li>
77   * </ul>
78   * If the Login Request is a leading Login Request, the target MUST use the value presented in CmdSN as the
79   * target value for ExpCmdSN.
80   * <p>
81   * <h4>ExpStatSN</h4> For the first Login Request on a connection this is ExpStatSN for the old connection and
82   * this field is only valid if the Login Request restarts a connection (see Section 5.3.4 Connection
83   * Reinstatement). For subsequent Login Requests it is used to acknowledge the Login Responses with their
84   * increasing StatSN values.
85   * <p>
86   * <h4>Login Parameters (Data Segment)</h4> The initiator MUST provide some basic parameters in order to
87   * enable the target to determine if the initiator may use the target’s resources and the initial text
88   * parameters for the security exchange. All the rules specified in Section 10.10.5 Text for text requests
89   * also hold for Login Requests. Keys and their explanations are listed in Chapter 11 (security negotiation
90   * keys) and Chapter 12 (operational parameter negotiation keys). All keys in Chapter 12, except for the X
91   * extension formats, MUST be supported by iSCSI initiators and targets. Keys in Chapter 11 only need to be
92   * supported when the function to which they refer is mandatory to implement.
93   * <p>
94   * 
95   * @author Volker Wildi
96   */

446     /** {@inheritDoc} */
447     @Override
448     protected final void deserializeBytes1to3(final int line) throws InternetSCSIException {
449 
450         continueFlag = Utils.isBitSet(line & Constants.CONTINUE_FLAG_MASK);
451 
452         Utils.isReserved(line & LoginConstants.BIT_11_AND_12_FLAG_MASK);
453 
454         currentStageNumber =
455             LoginStage
456                 .valueOf((byte)((line & LoginConstants.CSG_FLAG_MASK) >>> LoginConstants.CSG_BIT_SHIFT));
457         nextStageNumber =
458             LoginStage.valueOf((byte)((line & LoginConstants.NSG_FLAG_MASK) >>> Constants.TWO_BYTES_SHIFT));
459 
460         maxVersion = (byte)((line & Constants.THIRD_BYTE_MASK) >> Constants.ONE_BYTE_SHIFT);
461         minVersion = (byte)(line & Constants.FOURTH_BYTE_MASK);
462     }

464     /** {@inheritDoc} */
465     @Override
466     protected final void deserializeBytes12to15(final int line) throws InternetSCSIException {
467 
468         // use the logicalUnitNumber variable as temporary storage
469         final long l = Utils.getUnsignedLong(line);
470 
471         logicalUnitNumber |= l;
472         initiatorSessionID.deserialize(logicalUnitNumber);

474         targetSessionIdentifyingHandle = (short)(line & Constants.LAST_TWO_BYTES_MASK);

479     protected void deserializeBytes20to23(final int line) throws InternetSCSIException {

488     /** {@inheritDoc} */
489     @Override
490     protected final void checkIntegrity() throws InternetSCSIException {
491 
492         String exceptionMessage;
493 
494         do {
495             final BasicHeaderSegment bhs = protocolDataUnit.getBasicHeaderSegment();
496             if (bhs.isFinalFlag() && continueFlag) {
497                 exceptionMessage = "Transit and Continue Flag cannot be set at the same time.";
498                 break;
499             }
500 
501             if (!bhs.isFinalFlag() && nextStageNumber != LoginStage.SECURITY_NEGOTIATION) {
502                 exceptionMessage = "NextStageNumber is reserved, when the TransitFlag is not set.";
503                 break;
504             }
505 
506             if (bhs.isFinalFlag()) {
507                 if (currentStageNumber == LoginStage.SECURITY_NEGOTIATION) {
508                     if (nextStageNumber == LoginStage.SECURITY_NEGOTIATION) {
509                         exceptionMessage = "This transition (SNP -> SNP) is not allowed.";
510                         break;
511                     }
512                 } else if (currentStageNumber == LoginStage.LOGIN_OPERATIONAL_NEGOTIATION) {
513                     if (nextStageNumber == LoginStage.SECURITY_NEGOTIATION) {
514                         exceptionMessage = "This transition (LONP -> SNP) is not allowed.";
515                         break;
516                     } else if (nextStageNumber == LoginStage.LOGIN_OPERATIONAL_NEGOTIATION) {
517                         exceptionMessage = "This transition (LONP -> LONP) is not allowed.";
518                         break;
519                     }
520                 }
521             }
522 
523             if (minVersion != 0x00) {
524                 exceptionMessage = "MinVersion is not in a valid range.";
525                 break;
526             }
527 
528             if (maxVersion != 0x00) {
529                 exceptionMessage = "MaxVersion is not in a valid range.";
530                 break;
531             }
532 
533             if (minVersion != maxVersion) {
534                 exceptionMessage = "MinVersion and MaxVersion have not the same value.";
535                 break;
536             }
537 
538             // message is checked correctly
539             return;
540         } while (false);
541 
542         throw new InternetSCSIException(exceptionMessage);
543     }

548     /** {@inheritDoc} */
549     @Override
550     protected final int serializeBytes1to3() {
551 
552         int line = minVersion;
553         line |= maxVersion << Constants.ONE_BYTE_SHIFT;
554 
555         line |= nextStageNumber.value() << Constants.TWO_BYTES_SHIFT;
556         line |= currentStageNumber.value() << LoginConstants.CSG_BIT_SHIFT;
557 
558         if (continueFlag) {
559             line |= Constants.CONTINUE_FLAG_MASK;
560         }
561 
562         return line;
563     }
