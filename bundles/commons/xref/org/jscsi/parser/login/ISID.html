38   * <h1>ISID</h1>
39   * <p>
40   * This is an initiator-defined component of the session identifier and is structured as follows (see
41   * [RFC3721] and Section 9.1.1 Conservative Reuse of ISIDs for details):
42   * <p>
43   * <table border="1">
44   * <tr>
45   * <th>Byte</th>
46   * <th colspan="8">0</th>
47   * <th colspan="8">1</th>
48   * <th * colspan="8">2</th>
49   * <th colspan="8">3</th>
50   * </tr>
51   * <tr>
52   * <th>Bits</th>
53   * <td>0</td>
54   * <td>1</td>
55   * <td>2</td>
56   * <td>3</td>
57   * <td>4</td>
58   * <td>5</td>
59   * <td>6</td>
60   * <td>7</td>
61   * <td>0</td>
62   * <td>1</td>
63   * <td>2</td>
64   * <td>3</td>
65   * <td>4</td>
66   * <td>5</td>
67   * <td>6</td>
68   * <td>7</td>
69   * <td>0</td>
70   * <td>1</td>
71   * <td>2</td>
72   * <td>3</td>
73   * <td>4</td>
74   * <td>5</td>
75   * <td>6</td>
76   * <td>7</td>
77   * <td>0</td>
78   * <td>1</td>
79   * <td>2</td>
80   * <td>3</td>
81   * <td>4</td>
82   * <td>5</td>
83   * <td>6</td>
84   * <td>7</td>
85   * </tr>
86   * <tr>
87   * <td>8</td>
88   * <td colspan="2"><center>T</center></td>
89   * <td colspan="6"><center>A</center></td>
90   * <td colspan="16"><center>B</center></td>
91   * <td colspan="8"><center>C</center></td>
92   * </tr>
93   * <tr>
94   * <td>12</td>
95   * <td colspan="16"><center>D</center></td>
96   * <td colspan="16"/>
97   * </tr>
98   * </table>
99   * <p>
100  * The T field identifies the format and usage of A, B, C, and D as indicated below:
101  * <p>
102  * <table border="1">
103  * <tr>
104  * <th>T</th>
105  * <th>Description</th>
106  * </tr>
107  * <tr>
108  * <td>00b</td>
109  * <td>OUI-Format<br/>
110  * A&B are a <code>22</code> bit OUI (the I/G & U/L bits are omitted)<br/>
111  * C&D 24 bit qualifier</td>
112  * </tr>
113  * <tr>
114  * <td>01b</td>
115  * <td>EN - Format (IANA Enterprise Number)<br/>
116  * A - Reserved<br/>
117  * B&C EN (IANA Enterprise Number)<br/>
118  * D - Qualifier</td>
119  * </tr>
120  * <tr>
121  * <td>10b</td>
122  * <td>"Random"<br/>
123  * A - Reserved<br/>
124  * B&C Random<br/>
125  * D - Qualifier</td>
126  * </tr>
127  * <tr>
128  * <td>11b</td>
129  * <td>A,B,C&D Reserved</td>
130  * </tr>
131  * </table>
132  * <p>
133  * For the <code>T</code> field values <code>00b</code> and <code>01b</code>, a combination of <code>A</code>
134  * and <code>B</code> (for <code>00b</code>) or <code>B</code> and <code>C</code> (for <code>01b</code>)
135  * identifies the vendor or organization whose component (software or hardware) generates this ISID. A vendor
136  * or organization with one or more OUIs, or one or more Enterprise Numbers, MUST use at least one of these
137  * numbers and select the appropriate value for the <code>T</code> field when its components generate ISIDs.
138  * An <code>OUI</code> or <code>EN</code> MUST be set in the corresponding fields in network byte order (byte
139  * big-endian).
140  * <p>
141  * If the <code>T</code> field is <code>10b</code>, <code>B</code> and <code>C</code> are set to a random
142  * <code>24</code>-bit unsigned integer value in network byte order (byte big-endian). See [RFC3721] for how
143  * this affects the principle of "conservative reuse".
144  * <p>
145  * The Qualifier field is a <code>16</code> or <code>24</code>-bit unsigned integer value that provides a
146  * range of possible values for the ISID within the selected namespace. It may be set to any value within the
147  * constraints specified in the iSCSI protocol (see Section 3.4.3 Consequences of the Model and Section 9.1.1
148  * Conservative Reuse of ISIDs).
149  * <p>
150  * The <code>T</code> field value of <code>11b</code> is reserved.
151  * <p>
152  * If the ISID is derived from something assigned to a hardware adapter or interface by a vendor, as a preset
153  * default value, it MUST be configurable to a value assigned according to the SCSI port behavior desired by
154  * the system in which it is installed (see Section 9.1.1 Conservative Reuse of ISIDs and Section 9.1.2 iSCSI
155  * Name, ISID, and TPGT Use). The resultant ISID MUST also be persistent over power cycles, reboot, card swap,
156  * etc. For details have a look in the [RFC3721].
157  * 
158  * @author Volker Wildi
159  */

166      * Enumerations of all valid formats defined in the T field.

168     static enum Format {
169         /** ISID is in the Organization Unique Identifier Format. */
170         OUI_FORMAT((byte)0),
171         /** ISID is in the EN-Format (IANA Enterprise Number). */
172         IANA_ENTERPRISE_NUMBER((byte)1),
173         /** ISID is in the "Random" Format. */
174         RANDOM((byte)2),
175         /** ISID is in the Reserved. */
176         RESERVED((byte)3);
177 
178         private final byte value;
179 
180         private static Map<Byte, Format> mapping;
181 
182         static {
183             Format.mapping = new HashMap<Byte, Format>();
184             for (Format s : values()) {
185                 Format.mapping.put(s.value, s);
186             }
187         }
188 
189         private Format(final byte newValue) {
190 
191             value = newValue;
192         }

303     /**
304      * Serializes this ISID object ot its byte representation.
305      * 
306      * @return The byte representation of this ISID object.
307      * @throws InternetSCSIException
308      *             If any violation of the iSCSI-Standard emerge.
309      */
310     public final long serialize() throws InternetSCSIException {
311 
312         checkIntegrity();
313 
314         long isid = 0;
315         int firstLine = c;
316 
317         firstLine |= b << Constants.ONE_BYTE_SHIFT;
318         firstLine |= a << Constants.THREE_BYTES_SHIFT;
319         firstLine |= t.value() << T_FIELD_SHIFT;
320 
321         isid = Utils.getUnsignedLong(firstLine) << Constants.FOUR_BYTES_SHIFT;
322         isid |= Utils.getUnsignedLong(d) << Constants.TWO_BYTES_SHIFT;
323 
324         return isid;
325     }

327     /**
328      * Parses a given ISID in this ISID obejct.
329      * 
330      * @param isid
331      *            The byte representation of a ISID to parse.
332      * @throws InternetSCSIException
333      *             If any violation of the iSCSI-Standard emerge.
334      */
335     final void deserialize(final long isid) throws InternetSCSIException {
336 
337         int line = (int)((isid & FIRST_LINE_FLAG_MASK) >>> Constants.FOUR_BYTES_SHIFT);
338 
339         t = Format.valueOf((byte)(line >>> T_FIELD_SHIFT));
340         a = (byte)(line & A_FIELD_FLAG_MASK >>> Constants.THREE_BYTES_SHIFT);
341         b = (short)((line & Constants.MIDDLE_TWO_BYTES_SHIFT) >>> Constants.ONE_BYTE_SHIFT);
342         c = (byte)(line & Constants.FOURTH_BYTE_MASK);
343 
344         line = (int)(isid & Constants.LAST_FOUR_BYTES_MASK);
345         d = (short)((line & Constants.FIRST_TWO_BYTES_MASK) >>> Constants.TWO_BYTES_SHIFT);
346 
347         checkIntegrity();
348     }

485     /**
486      * This method checks, if all fields are valid. In these cases an exception
487      * will be thrown.
488      * 
489      * @throws InternetSCSIException
490      *             If the integrity is violated.
491      */
492     protected final void checkIntegrity() throws InternetSCSIException {
493 
494         String exceptionMessage = "";
495         switch (t) {
496         case OUI_FORMAT:
497             break;
498 
499         case IANA_ENTERPRISE_NUMBER:
500             break;
501 
502         case RANDOM:
503             if (d != 0) {
504                 exceptionMessage = "The D field is reserved in this ISID Format.";
505             }
506             break;
507 
508         case RESERVED:
509             if (a != 0 && b != 0 && c != 0 && d != 0) {
510                 exceptionMessage = "This ISID is not valid. All";
511             }
512             break;
513 
514         default:
515             exceptionMessage = "This format is not supported.";
516         }
517 
518         if (exceptionMessage.length() > 0) {
519             throw new InternetSCSIException(exceptionMessage);
520         } else {
521             // no error occured... Nice! :-)
522         }
523     }
